    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <!-- Iconos Lucide -->
    <script src="https://unpkg.com/lucide@latest"></script>
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;700;900&family=Playfair+Display:wght@700&family=JetBrains+Mono&display=swap');

        .animate-in { animation: fadeIn 0.3s ease-out; }

<body class="bg-slate-50">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useRef, useMemo } = React;

        // Ayudante para Iconos Lucide
        const Icon = ({ name, size = 24, className = "" }) => {
            useEffect(() => {
                if (window.lucide) window.lucide.createIcons();
            }, [name]);
            return <i data-lucide={name} style={{ width: size, height: size }} className={`inline-block ${className}`}></i>;
        };

        function App() {
            // --- ESTADOS DE DATOS ---
            const [songs, setSongs] = useState([
                { id: '1', title: 'Cuan Grande es Él', artist: 'Himno Clásico', musicalKey: 'Sol', lyrics: '[Sol]Señor mi Dios, al con[Do]templar los cielos\nEl [Sol]firmamento y las [Re]estrellas mil\n[Sol]Al oir tu voz en [Do]los potentes truenos\nY [Sol]ver brillar al [Re]sol en su cenit.' },
                { id: '2', title: 'Agnus Dei', artist: 'Michael W. Smith', musicalKey: 'La', lyrics: '[La]Aleluya, [Re]Aleluya\nReinas [La]Tú, poderoso Dios\n[Re]Aleluya, [La]Aleluya\n[Sim]Santo, [La]Santo\n[Re]Eres [Mi]Tú.' }
            ]);
            const [playlists, setPlaylists] = useState([
                { id: 'p1', name: 'Culto de Adoración', songs: ['1', '2'], updatedAt: Date.now() }
            ]);

            // --- ESTADOS DE UI ---
            const [currentView, setCurrentView] = useState('list'); 
            const [activeTab, setActiveTab] = useState('songs');
            const [activeSongId, setActiveSongId] = useState(null);
            const [activePlaylistId, setActivePlaylistId] = useState(null);
            const [searchQuery, setSearchQuery] = useState('');
            const [playlistMode, setPlaylistMode] = useState('list'); 
            const [isDarkMode, setIsDarkMode] = useState(false);
            const [showStylePanel, setShowStylePanel] = useState(false);
            const [showAddSongModal, setShowAddSongModal] = useState(false);
            const [showNewPlaylistModal, setShowNewPlaylistModal] = useState(false);
            const [newPlaylistName, setNewPlaylistName] = useState('');
            
            const [textSize, setTextSize] = useState(20);
            const [fontFamily, setFontFamily] = useState('font-sans');
            const [lyricColor, setLyricColor] = useState('default');
            const [chordColor, setChordColor] = useState('default');

            const [isScrolling, setIsScrolling] = useState(false);
            const [scrollSpeed, setScrollSpeed] = useState(2);
            const [bpm, setBpm] = useState(120);
            const [isMetronomePlaying, setIsMetronomePlaying] = useState(false);
            const [transposeSteps, setTransposeSteps] = useState(0);
            const [isAiLoading, setIsAiLoading] = useState(false);
            const [toast, setToast] = useState(null);

            const [editTitle, setEditTitle] = useState('');
            const [editArtist, setEditArtist] = useState('');
            const [editKey, setEditKey] = useState('');
            const [editLyrics, setEditLyrics] = useState('');

            const audioCtxRef = useRef(null);
            const metronomeIntervalRef = useRef(null);

            // --- LÓGICA MUSICAL ---
            const transposeChord = (chordStr, steps) => {
                if (!chordStr || steps === 0) return chordStr;
                const notes = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
                const map = { 'Db':'C#', 'Eb':'D#', 'Gb':'F#', 'Ab':'G#', 'Bb':'A#', 'Do':'C', 'Re':'D', 'Mi':'E', 'Fa':'F', 'Sol':'G', 'La':'A', 'Si':'B' };
                return chordStr.replace(/([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?/gi, (m) => {
                    let note = m.charAt(0).toUpperCase() + m.slice(1).toLowerCase();
                    if (map[note]) note = map[note];
                    let idx = notes.indexOf(note);
                    if (idx === -1) return m;
                    let newIdx = (idx + steps) % 12;
                    if (newIdx < 0) newIdx += 12;
                    return notes[newIdx];
                });
            };

            const lyricStyleMap = { 'default': isDarkMode ? 'text-slate-200' : 'text-slate-800', 'blue': 'text-blue-600', 'emerald': 'text-emerald-600', 'rose': 'text-rose-600', 'amber': 'text-amber-600' };
            const chordStyleMap = { 'default': 'text-indigo-600', 'red': 'text-red-600', 'teal': 'text-teal-600', 'violet': 'text-violet-600', 'orange': 'text-orange-600' };

            const renderLyrics = (text, steps = 0) => {
                if (!text) return null;
                return text.split('\n').map((line, i) => (
                    <div key={i} className={`flex flex-wrap items-end mb-1 ${fontFamily}`}>
                        {line.split(/(\[[^\]]+\])/g).map((part, j) => {
                            if (part.startsWith('[') && part.endsWith(']')) {
                                return (
                                    <div key={j} className="flex flex-col mr-1">
                                        <span className={`font-bold block -mb-1 ${chordStyleMap[chordColor]} print:text-blue-800`} style={{ fontSize: `${textSize * 0.7}px` }}>
                                            {transposeChord(part.slice(1, -1), steps)}
                                        </span>
                                    </div>
                                );
                            }
                            return <span key={j} className={`${lyricStyleMap[lyricColor]} print:text-black whitespace-pre leading-relaxed`} style={{ fontSize: `${textSize}px` }}>{part}</span>;
                        })}
                    </div>
                ));
            };

            const showToast = (msg) => { setToast(msg); setTimeout(() => setToast(null), 3000); };

            // --- IA GEMINI ---
            const askGemini = async (prompt) => {
                setIsAiLoading(true);
                const apiKey = "gen-lang-client-0332007452";
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
                    });
                    const data = await response.json();
                    return data.candidates?.[0]?.content?.parts?.[0]?.text || null;
                } catch (e) { return null; }
                finally { setIsAiLoading(false); }
            };

            const handleRearmonizar = async () => {
                const r = await askGemini(`ACTÚA COMO UN PROCESADOR DE TEXTO. Re-armoniza con acordes avanzados para piano en corchetes [Acorde]. SIN COMENTARIOS. Letra: ${editLyrics}`);
                if (r) { setEditLyrics(r.replace(/```/g, '').trim()); showToast("¡Rearmonización lista!"); }
            };

            // --- METRÓNOMO ---
            useEffect(() => {
                if (isMetronomePlaying) {
                    metronomeIntervalRef.current = setInterval(() => {
                        if (!audioCtxRef.current) audioCtxRef.current = new (window.AudioContext || window.webkitAudioContext)();
                        const osc = audioCtxRef.current.createOscillator();
                        const gain = audioCtxRef.current.createGain();
                        osc.connect(gain); gain.connect(audioCtxRef.current.destination);
                        osc.frequency.value = 1000;
                        gain.gain.setValueAtTime(0.2, audioCtxRef.current.currentTime);
                        gain.gain.exponentialRampToValueAtTime(0.001, audioCtxRef.current.currentTime + 0.05);
                        osc.start(); osc.stop(audioCtxRef.current.currentTime + 0.05);
                    }, 60000 / bpm);
                } else { clearInterval(metronomeIntervalRef.current); }
                return () => clearInterval(metronomeIntervalRef.current);
            }, [isMetronomePlaying, bpm]);

            // --- AUTO SCROLL ---
            useEffect(() => {
                let interval;
                if (isScrolling) interval = setInterval(() => window.scrollBy(0, scrollSpeed), 50);
                return () => clearInterval(interval);
            }, [isScrolling, scrollSpeed]);

            const activeSong = songs.find(s => s.id === activeSongId);
            const activePlaylist = playlists.find(p => p.id === activePlaylistId);

            const filteredSongs = songs.filter(s => 
                s.title.toLowerCase().includes(searchQuery.toLowerCase()) || 
                s.artist.toLowerCase().includes(searchQuery.toLowerCase()) ||
                s.musicalKey.toLowerCase().includes(searchQuery.toLowerCase())
            );

            return (
                <div className={`min-h-screen transition-colors duration-300 ${isDarkMode ? 'bg-slate-950 text-white' : 'bg-slate-50 text-slate-900'}`}>
                    
                    {toast && <div className="fixed top-6 left-1/2 -translate-x-1/2 bg-indigo-600 text-white px-6 py-3 rounded-full font-bold shadow-2xl z-[100] animate-in">{toast}</div>}

                    <header className="sticky top-0 z-50 bg-indigo-700 text-white p-4 shadow-lg flex justify-between items-center print:hidden">
                        <div className="flex items-center gap-2 cursor-pointer" onClick={() => {setCurrentView('list'); setIsScrolling(false)}}>
                            <Icon name="music" />
                            <h1 className="text-xl font-black italic tracking-tighter leading-none">CORARIO AI</h1>
                        </div>
                        <div className="flex items-center gap-4">
                            <button onClick={() => setIsDarkMode(!isDarkMode)}><Icon name={isDarkMode ? "sun" : "moon"} size={20} /></button>
                            {currentView === 'list' && (
                                <button className="bg-white text-indigo-700 px-4 py-1.5 rounded-full font-bold text-sm shadow-md" onClick={() => {setEditTitle(''); setEditArtist(''); setEditLyrics(''); setEditKey(''); setActiveSongId(null); setCurrentView('edit');}}>+ Nueva</button>
                            )}
                        </div>
                    </header>

                    <main className="max-w-4xl mx-auto p-4 md:p-8">
                        {currentView === 'list' && (
                            <div className="animate-in">
                                <div className="flex gap-2 p-1 bg-slate-200 dark:bg-slate-800 rounded-2xl mb-8">
                                    <button onClick={() => setActiveTab('songs')} className={`flex-1 py-3 rounded-xl font-bold transition-all ${activeTab === 'songs' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500'}`}>Alabanzas</button>
                                    <button onClick={() => setActiveTab('playlists')} className={`flex-1 py-3 rounded-xl font-bold transition-all ${activeTab === 'playlists' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500'}`}>Cultos</button>
                                </div>

                                <div className="relative mb-8">
                                    <Icon name="search" className="absolute left-4 top-3.5 opacity-30" size={20} />
                                    <input 
                                        type="text" 
                                        placeholder="Buscar por título, artista o tono..."
                                        className="w-full pl-12 pr-4 py-3.5 rounded-2xl border-2 dark:bg-slate-900 border-transparent focus:border-indigo-500 outline-none shadow-sm transition-all"
                                        value={searchQuery}
                                        onChange={(e) => setSearchQuery(e.target.value)}
                                    />
                                </div>

                                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                                    {activeTab === 'songs' ? (
                                        filteredSongs.map(s => (
                                            <div key={s.id} onClick={() => {setActiveSongId(s.id); setTransposeSteps(0); setCurrentView('view')}} className="p-5 bg-white dark:bg-slate-900 rounded-3xl border-2 hover:border-indigo-500 cursor-pointer flex justify-between items-center shadow-sm transition-all">
                                                <div className="overflow-hidden"><h3 className="font-bold text-lg leading-tight truncate">{s.title}</h3><p className="text-xs opacity-50 uppercase tracking-wider">{s.artist} • [{s.musicalKey}]</p></div>
                                                <Icon name="chevron-right" className="opacity-20" />
                                            </div>
                                        ))
                                    ) : (
                                        <>
                                            <button onClick={() => setShowNewPlaylistModal(true)} className="p-8 rounded-3xl border-2 border-dashed border-slate-300 flex flex-col items-center justify-center gap-2 text-slate-400 hover:bg-indigo-50 transition-all"><Icon name="folder-plus" /><span className="font-bold">Nuevo Culto</span></button>
                                            {playlists.map(p => (
                                                <div key={p.id} onClick={() => {setActivePlaylistId(p.id); setPlaylistMode('list'); setCurrentView('playlistView')}} className="p-6 bg-white dark:bg-slate-900 rounded-3xl border-2 border-dashed border-slate-300 hover:border-indigo-500 cursor-pointer shadow-sm group">
                                                    <div className="flex items-center gap-3 mb-2"><Icon name="list-music" className="text-indigo-500" /><h3 className="font-black text-xl">{p.name}</h3></div>
                                                    <p className="text-sm opacity-50">{p.songs.length} canciones preparadas</p>
                                                </div>
                                            ))}
                                        </>
                                    )}
                                </div>
                            </div>
                        )}

                        {currentView === 'view' && activeSong && (
                            <div className="animate-in pb-32">
                                <div className="flex justify-between items-center mb-8 print:hidden">
                                    <button className="flex items-center gap-2 text-indigo-600 font-bold" onClick={() => {setCurrentView('list'); setIsScrolling(false)}}><Icon name="chevron-left" size={18} /> Volver</button>
                                    <div className="flex gap-2">
                                        <button className="p-3 bg-indigo-100 text-indigo-700 rounded-2xl" onClick={() => setShowStylePanel(!showStylePanel)}><Icon name="palette" size={20}/></button>
                                        <button className="p-3 bg-slate-800 text-white rounded-2xl" onClick={() => window.print()}><Icon name="printer" size={20}/></button>
                                    </div>
                                </div>

                                {showStylePanel && (
                                    <div className="bg-white dark:bg-slate-900 p-6 rounded-3xl border-2 shadow-xl mb-8 animate-in print:hidden">
                                        <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                                            <div><label className="text-[10px] font-black uppercase text-slate-400 block mb-3">Color Letra</label><div className="flex gap-2">{['default', 'blue', 'emerald', 'rose', 'amber'].map(c => (<button key={c} onClick={()=>setLyricColor(c)} className={`w-8 h-8 rounded-full border-2 ${lyricColor===c?'border-indigo-500 scale-110 shadow-md':'border-transparent'}`} style={{backgroundColor: c==='default'?'gray':c==='emerald'?'#10b981':c==='rose'?'#f43f5e':c==='amber'?'#f59e0b':'#3b82f6'}}></button>))}</div></div>
                                            <div><label className="text-[10px] font-black uppercase text-slate-400 block mb-3">Color Acordes</label><div className="flex gap-2">{['default', 'red', 'teal', 'violet', 'orange'].map(c => (<button key={c} onClick={()=>setChordColor(c)} className={`w-8 h-8 rounded-full border-2 ${chordColor===c?'border-indigo-500 scale-110 shadow-md':'border-transparent'}`} style={{backgroundColor: c==='default'?'#6366f1':c==='red'?'#ef4444':c==='teal'?'#14b8a6':c==='violet'?'#8b5cf6':'#f97316'}}></button>))}</div></div>
                                        </div>
                                    </div>
                                )}

                                <div className="mb-10">
                                    <h2 className={`text-5xl font-black mb-2 ${fontFamily}`}>{activeSong.title}</h2>
                                    <p className="text-xl opacity-60 italic mb-6">{activeSong.artist}</p>
                                    <div className="flex flex-wrap gap-4 print:hidden">
                                        <div className="flex items-center bg-white dark:bg-slate-800 rounded-2xl border-2 p-1">
                                            <button className="w-10 h-10 font-black" onClick={() => setTransposeSteps(s => s - 1)}>-</button>
                                            <span className="px-4 text-indigo-600 font-black">{transposeSteps !== 0 ? (transposeSteps > 0 ? `+${transposeSteps}` : transposeSteps) : 'Tono Original'}</span>
                                            <button className="w-10 h-10 font-black" onClick={() => setTransposeSteps(s => s + 1)}>+</button>
                                        </div>
                                        <div className="flex items-center bg-white dark:bg-slate-800 rounded-2xl border-2 p-1">
                                            <button className="px-3" onClick={() => setBpm(b => Math.max(40, b - 5))}>-</button>
                                            <span className="font-bold text-xs w-16 text-center">{bpm} BPM</span>
                                            <button className="px-3" onClick={() => setBpm(b => Math.min(240, b + 5))}>+</button>
                                            <button className={`ml-2 p-2 rounded-xl ${isMetronomePlaying?'bg-red-500':'bg-teal-500'} text-white`} onClick={() => setIsMetronomePlaying(!isMetronomePlaying)}><Icon name={isMetronomePlaying ? "pause" : "play"} size={16}/></button>
                                        </div>
                                    </div>
                                </div>
                                <div className="space-y-1">{renderLyrics(activeSong.lyrics, transposeSteps)}</div>
                            </div>
                        )}

                        {currentView === 'playlistView' && activePlaylist && (
                            <div className="animate-in pb-40">
                                <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-10 gap-4 border-b pb-6">
                                    <div><button onClick={() => setCurrentView('list')} className="text-indigo-600 font-bold flex items-center gap-1 mb-2"><Icon name="chevron-left" size={16}/> Volver</button><h2 className="text-4xl font-black">{activePlaylist.name}</h2></div>
                                    <div className="flex gap-2 w-full md:w-auto">
                                        <button onClick={() => { setPlaylistMode(playlistMode === 'list' ? 'full' : 'list'); setIsScrolling(false); }} className={`flex-1 md:flex-none px-6 py-3 rounded-2xl font-bold flex items-center justify-center gap-2 shadow-lg ${playlistMode === 'full' ? 'bg-indigo-600 text-white' : 'bg-indigo-100 text-indigo-700'}`}><Icon name={playlistMode === 'list' ? "eye" : "layout-list"} size={20} /> {playlistMode === 'list' ? "Modo Presentación" : "Ver Lista"}</button>
                                        <button onClick={()=>setShowAddSongModal(true)} className="p-3 bg-indigo-600 text-white rounded-2xl shadow-lg"><Icon name="plus"/></button>
                                    </div>
                                </div>
                                {playlistMode === 'list' ? (
                                    <div className="space-y-4">{activePlaylist.songs.map((sid, idx) => {
                                        const s = songs.find(x => x.id === sid);
                                        if (!s) return null;
                                        return (
                                            <div key={idx} className="flex items-center p-5 bg-white dark:bg-slate-900 rounded-3xl border-2 shadow-sm">
                                                <div className="w-10 h-10 rounded-full bg-indigo-100 text-indigo-600 flex items-center justify-center font-black mr-4">{idx+1}</div>
                                                <div className="flex-1 cursor-pointer" onClick={() => {setActiveSongId(s.id); setCurrentView('view')}}><h4 className="font-bold text-lg">{s.title}</h4><p className="text-sm opacity-50">{s.artist}</p></div>
                                                <button onClick={()=>{
                                                    const news = activePlaylist.songs.filter(id => id !== sid);
                                                    setPlaylists(playlists.map(p => p.id === activePlaylist.id ? {...p, songs: news} : p));
                                                }} className="p-2 text-red-500"><Icon name="trash-2" size={18}/></button>
                                            </div>
                                        );
                                    })}</div>
                                ) : (
                                    <div className="space-y-40">{activePlaylist.songs.map((sid, idx) => {
                                        const s = songs.find(x => x.id === sid);
                                        if (!s) return null;
                                        return (
                                            <div key={idx} className="song-page animate-in">
                                                <div className="mb-10 border-b-4 border-indigo-600/20 pb-4"><span className="text-7xl font-black text-indigo-600/10 block mb-2">{idx + 1}</span><h3 className={`text-4xl font-black ${fontFamily}`}>{s.title}</h3><p className="text-xl opacity-60">{s.artist} • [{s.musicalKey}]</p></div>
                                                <div className="text-2xl">{renderLyrics(s.lyrics, 0)}</div>
                                            </div>
                                        );
                                    })}</div>
                                )}
                            </div>
                        )}

                        {currentView === 'edit' && (
                            <div className="animate-in pb-20">
                                <div className="flex justify-between items-center mb-8"><h2 className="text-3xl font-black">Editor de Alabanza</h2><div className="flex gap-2"><button onClick={() => setCurrentView('list')} className="px-4 py-2 font-bold opacity-50">Cancelar</button><button onClick={handleRearmonizar} className="bg-indigo-100 text-indigo-700 px-6 py-2 rounded-xl font-bold flex items-center gap-2" disabled={isAiLoading}>{isAiLoading ? <Icon name="loader-2" className="animate-spin" /> : <Icon name="sparkles" size={18} />} IA Rearmonizar</button></div></div>
                                <div className="space-y-4">
                                    <input type="text" placeholder="Título..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 text-xl font-bold" value={editTitle} onChange={e => setEditTitle(e.target.value)} />
                                    <div className="grid grid-cols-2 gap-4"><input type="text" placeholder="Artista..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-bold" value={editArtist} onChange={e => setEditArtist(e.target.value)} /><input type="text" placeholder="Tono..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-bold" value={editKey} onChange={e => setEditKey(e.target.value)} /></div>
                                    <textarea placeholder="Pega letra con acordes [Do]..." className="w-full h-96 p-6 rounded-3xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-mono text-sm leading-relaxed" value={editLyrics} onChange={e => setEditLyrics(e.target.value)} />
                                    <button className="w-full py-5 bg-indigo-600 text-white rounded-3xl font-black text-xl shadow-xl hover:bg-indigo-700" onClick={() => {
                                        const newSong = { id: Date.now().toString(), title: editTitle || 'Sin título', artist: editArtist || 'Autor', musicalKey: editKey || 'S/D', lyrics: editLyrics };
                                        setSongs([...songs, newSong]); setCurrentView('list'); showToast("¡Guardada!");
                                    }}>GUARDAR ALABANZA</button>
                                </div>
                            </div>
                        )}

                        {/* CONTROLES FLOTANTES */}
                        {(currentView === 'view' || (currentView === 'playlistView' && playlistMode === 'full')) && (
                            <div className="fixed bottom-8 right-8 flex items-center gap-3 p-3 bg-white/90 dark:bg-slate-800/90 backdrop-blur rounded-full shadow-2xl border-2 print:hidden z-[100]">
                                {isScrolling && (<div className="flex items-center gap-2 pr-2 animate-in slide-in-from-right-4"><button className="p-3 bg-slate-100 dark:bg-slate-700 rounded-full" onClick={() => setScrollSpeed(s => Math.max(1, s-1))}><Icon name="rewind" size={18}/></button><span className="font-black text-indigo-600 px-1 text-sm">{scrollSpeed}x</span><button className="p-3 bg-slate-100 dark:bg-slate-700 rounded-full" onClick={() => setScrollSpeed(s => Math.min(10, s+1))}><Icon name="fast-forward" size={18}/></button></div>)}
                                <button onClick={() => setIsScrolling(!isScrolling)} className={`p-6 rounded-full text-white shadow-xl transition-all ${isScrolling ? 'bg-red-500' : 'bg-indigo-600'}`}><Icon name={isScrolling ? "pause" : "play"} size={32} /></button>
                            </div>
                        )}

                        {/* MODALES */}
                        {showAddSongModal && activePlaylist && (
                            <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-[100]"><div className="bg-white dark:bg-slate-800 rounded-3xl p-6 max-w-md w-full shadow-2xl max-h-[80vh] flex flex-col"><div className="flex justify-between items-center mb-4"><h3 className="text-xl font-bold">Añadir a {activePlaylist.name}</h3><button onClick={()=>setShowAddSongModal(false)}><Icon name="x" opacity="0.5"/></button></div><div className="overflow-y-auto space-y-2 flex-1 pr-1 custom-scrollbar">{songs.map(s => (<button key={s.id} onClick={() => { if (!activePlaylist.songs.includes(s.id)) { const news = [...activePlaylist.songs, s.id]; setPlaylists(playlists.map(p => p.id === activePlaylist.id ? {...p, songs: news} : p)); showToast("Agregada"); } }} className={`w-full p-4 rounded-xl border-2 text-left flex justify-between items-center ${activePlaylist.songs.includes(s.id) ? 'bg-green-50 border-green-200' : 'hover:bg-indigo-50'}`}><div><p className="font-bold">{s.title}</p><p className="text-xs opacity-50">{s.artist}</p></div>{activePlaylist.songs.includes(s.id) ? <Icon name="check" className="text-green-600" /> : <Icon name="plus" className="text-indigo-600" />}</button>))}</div><button onClick={()=>setShowAddSongModal(false)} className="w-full mt-4 py-4 bg-indigo-600 text-white rounded-xl font-bold">Hecho</button></div></div>
                        )}

                        {showNewPlaylistModal && (
                            <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-[100]"><div className="bg-white dark:bg-slate-800 rounded-3xl p-6 max-w-sm w-full shadow-2xl"><h3 className="text-xl font-bold mb-4">Nombre del Culto</h3><input type="text" className="w-full p-4 rounded-2xl border-2 mb-6 dark:bg-slate-900 outline-none" placeholder="Ej: Domingo Mañana" value={newPlaylistName} onChange={e=>setNewPlaylistName(e.target.value)} autoFocus /><div className="flex gap-3"><button onClick={()=>setShowNewPlaylistModal(false)} className="flex-1 py-3 font-bold opacity-50">Cancelar</button><button onClick={()=>{ if(newPlaylistName){ setPlaylists([...playlists, {id: Date.now().toString(), name: newPlaylistName, songs: [], updatedAt: Date.now()}]); setShowNewPlaylistModal(false); setNewPlaylistName(''); showToast("Culto Creado"); } }} className="flex-1 py-3 bg-indigo-600 text-white rounded-xl font-bold">Crear</button></div></div></div>
                        )}
                    </main>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
