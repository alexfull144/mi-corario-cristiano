import React, { useState, useEffect, useRef, useMemo } from 'react';
import { 
  Book, Plus, Music, Edit2, Trash2, ChevronLeft, Save, ZoomIn, ZoomOut, 
  Search, Wand2, Loader2, Cloud, Moon, Sun, ListMusic, FolderPlus, 
  ListPlus, X, Share2, Play, Pause, Rewind, FastForward, Link as LinkIcon, 
  Copy, Sparkles, BookOpen, Printer, Activity, Palette, Type, LayoutList, Eye 
} from 'lucide-react';

// --- CONFIGURACIÓN DE FIREBASE ---
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, collection, onSnapshot, addDoc, updateDoc, deleteDoc, doc, getDoc } from 'firebase/firestore';

const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

export default function App() {
  // --- ESTADOS DE DATOS ---
  const [songs, setSongs] = useState([]);
  const [playlists, setPlaylists] = useState([]);
  const [currentView, setCurrentView] = useState('list'); 
  const [activeTab, setActiveTab] = useState('songs');
  const [activeSongId, setActiveSongId] = useState(null);
  const [activePlaylistId, setActivePlaylistId] = useState(null);
  const [searchQuery, setSearchQuery] = useState('');
  
  // --- CONFIGURACIÓN VISUAL ---
  const [playlistMode, setPlaylistMode] = useState('list'); 
  const [isDarkMode, setIsDarkMode] = useState(false);
  const [textSize, setTextSize] = useState(18);
  const [transposeSteps, setTransposeSteps] = useState(0);
  const [showStylePanel, setShowStylePanel] = useState(false);
  const [fontFamily, setFontFamily] = useState('font-sans');
  const [lyricColor, setLyricColor] = useState('default');
  const [chordColor, setChordColor] = useState('default');

  // --- HERRAMIENTAS (Scroll, Metrónomo) ---
  const [isScrolling, setIsScrolling] = useState(false);
  const [scrollSpeed, setScrollSpeed] = useState(2);
  const [bpm, setBpm] = useState(120);
  const [isMetronomePlaying, setIsMetronomePlaying] = useState(false);
  const audioCtxRef = useRef(null);
  const metronomeIntervalRef = useRef(null);

  // --- IA GEMINI ---
  const [isAiLoading, setIsAiLoading] = useState(false);
  const [aiAnalysis, setAiAnalysis] = useState(null);

  // --- FIREBASE Y UI ---
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [itemToDelete, setItemToDelete] = useState(null);
  const [showAddToPlaylistModal, setShowAddToPlaylistModal] = useState(false);
  const [showAddSongToCurrentPlaylistModal, setShowAddSongToCurrentPlaylistModal] = useState(false);
  const [sharingUrl, setSharingUrl] = useState('');
  const [isSharing, setIsSharing] = useState(false);
  const [importId, setImportId] = useState(null);
  const [toast, setToast] = useState(null);
  const [showNewPlaylistModal, setShowNewPlaylistModal] = useState(false);
  const [newPlaylistName, setNewPlaylistName] = useState('');

  // Editor
  const [editTitle, setEditTitle] = useState('');
  const [editArtist, setEditArtist] = useState('');
  const [editKey, setEditKey] = useState(''); 
  const [editLyrics, setEditLyrics] = useState('');

  // --- SINCRONIZACIÓN DE DATOS ---
  const activeSong = useMemo(() => songs.find(s => s.id === activeSongId), [songs, activeSongId]);
  const activePlaylist = useMemo(() => playlists.find(p => p.id === activePlaylistId), [playlists, activePlaylistId]);

  // Autenticación inicial
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // Carga de datos desde Firestore
  useEffect(() => {
    if (!user) return;
    const unsubSongs = onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'songs'), (snap) => {
      setSongs(snap.docs.map(d => ({ id: d.id, ...d.data() })));
      setLoading(false);
    });

    const unsubPlaylists = onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'playlists'), (snap) => {
      setPlaylists(snap.docs.map(d => ({ id: d.id, ...d.data() })));
    });

    const params = new URLSearchParams(window.location.search);
    const shareId = params.get('share');
    if (shareId) setImportId(shareId);

    return () => { unsubSongs(); unsubPlaylists(); };
  }, [user]);

  // --- MOTOR DEL METRÓNOMO ---
  const playMetronomeTick = () => {
    if (!audioCtxRef.current) audioCtxRef.current = new (window.AudioContext || window.webkitAudioContext)();
    const osc = audioCtxRef.current.createOscillator();
    const gain = audioCtxRef.current.createGain();
    osc.connect(gain); gain.connect(audioCtxRef.current.destination);
    osc.frequency.value = 1000;
    gain.gain.setValueAtTime(0.5, audioCtxRef.current.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtxRef.current.currentTime + 0.05);
    osc.start(); osc.stop(audioCtxRef.current.currentTime + 0.05);
  };

  useEffect(() => {
    if (isMetronomePlaying) metronomeIntervalRef.current = setInterval(playMetronomeTick, 60000 / bpm);
    else clearInterval(metronomeIntervalRef.current);
    return () => clearInterval(metronomeIntervalRef.current);
  }, [isMetronomePlaying, bpm]);

  // --- LÓGICA DE AUTO-SCROLL ---
  useEffect(() => {
    if (isScrolling && (currentView === 'view' || (currentView === 'playlistView' && playlistMode === 'full'))) {
      const interval = setInterval(() => window.scrollBy(0, scrollSpeed), 50);
      return () => clearInterval(interval);
    }
  }, [isScrolling, scrollSpeed, currentView, playlistMode]);

  const showToast = (message, type = 'success') => {
    setToast({ message, type });
    setTimeout(() => setToast(null), 3000);
  };

  // --- MOTOR DE TRANSPOSICIÓN ---
  const transposeChord = (chordStr, steps) => {
    if (!chordStr || steps === 0) return chordStr;
    const notes = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
    const esNotes = ['Do', 'Do#', 'Re', 'Re#', 'Mi', 'Fa', 'Fa#', 'Sol', 'Sol#', 'La', 'La#', 'Si'];
    const map = { 'Db':'C#', 'Eb':'D#', 'Gb':'F#', 'Ab':'G#', 'Bb':'A#', 'Reb':'Do#', 'Mib':'Re#', 'Solb':'Fa#', 'Lab':'Sol#', 'Sib':'La#' };
    return chordStr.replace(/([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?/gi, (match) => {
      let m = match.charAt(0).toUpperCase() + match.slice(1).toLowerCase();
      if (map[m]) m = map[m];
      let idx = notes.indexOf(m);
      let isEs = false;
      if (idx === -1) { idx = esNotes.indexOf(m); isEs = true; }
      if (idx === -1) return match;
      let newIdx = (idx + steps) % 12;
      if (newIdx < 0) newIdx += 12;
      return isEs ? esNotes[newIdx] : notes[newIdx];
    });
  };

  // --- LLAMADAS A IA GEMINI ---
  const askGemini = async (prompt) => {
    setIsAiLoading(true);
    const apiKey = "";
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
      });
      const data = await response.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text || null;
    } catch (e) { showToast("Error IA", "error"); return null; }
    finally { setIsAiLoading(false); }
  };

  const reharmonizeWithAI = async () => {
    const prompt = `ACTÚA COMO UN PROCESADOR DE TEXTO PURO. Re-armoniza la siguiente canción con acordes de piano enriquecidos.
    REGLAS: 1. Devuelve SOLAMENTE la letra con los acordes en corchetes [Acorde]. 2. PROHIBIDO explicaciones o saludos. 3. Respeta la estructura.
    Letra: \n${editLyrics}`;

    const r = await askGemini(prompt);
    if (r) { 
      const cleaned = r.replace(/```(text|markdown)?/g, '').replace(/```/g, '').trim();
      setEditLyrics(cleaned); showToast("¡Rearmonización aplicada!"); 
    }
  };

  const generateDevotional = async () => {
    const r = await askGemini(`Genera un devocional cristiano muy breve para músicos basado en "${activeSong.title}". Letra: ${activeSong.lyrics}. Incluye versículo, reflexión y oración corta.`);
    if (r) setAiAnalysis(r);
  };

  // --- ESTILOS VISUALES ---
  const lyricStyleOptions = { 
    'default': isDarkMode ? 'text-slate-200' : 'text-slate-800', 
    'blue': 'text-blue-600', 'emerald': 'text-emerald-600', 
    'rose': 'text-rose-600', 'amber': 'text-amber-600' 
  };
  const chordStyleOptions = { 
    'default': 'text-indigo-600', 'red': 'text-red-600', 
    'teal': 'text-teal-600', 'violet': 'text-violet-600', 'orange': 'text-orange-600' 
  };

  const renderContent = (text, steps = transposeSteps) => {
    if (!text) return null;
    return text.split('\n').map((line, idx) => {
      if (!line.trim()) return <div key={idx} className="h-4"></div>;
      const parts = line.split(/(\[[^\]]+\])/g);
      let segments = [];
      let pendingChord = "";

      parts.forEach(p => {
        if (p.startsWith('[') && p.endsWith(']')) pendingChord = p.slice(1, -1);
        else { segments.push({ chord: pendingChord, text: p }); pendingChord = ""; }
      });
      if (pendingChord) segments.push({ chord: pendingChord, text: "" });

      return (
        <div key={idx} className={`flex flex-wrap items-end mb-1 ${fontFamily} break-inside-avoid`}>
          {segments.map((seg, i) => (
            <div key={i} className="flex flex-col">
              <span className={`font-bold leading-none mb-0.5 ${chordStyleOptions[chordColor]} print:text-blue-800`} style={{ fontSize: `${textSize * 0.75}px` }}>
                {seg.chord ? transposeChord(seg.chord, steps) : ""}
              </span>
              <span className={`whitespace-pre-wrap ${lyricStyleOptions[lyricColor]} print:text-black`} style={{ fontSize: `${textSize}px` }}>
                {seg.text}
              </span>
            </div>
          ))}
        </div>
      );
    });
  };

  // --- ACCIONES ---
  const handlePrint = () => {
    setIsMetronomePlaying(false);
    setIsScrolling(false);
    setShowStylePanel(false);
    showToast("Preparando PDF para imprimir...");
    setTimeout(() => {
      window.print();
    }, 500);
  };

  const saveSong = async () => {
    if (!editTitle) return showToast("Falta el título", "error");
    setSaving(true);
    const data = { title: editTitle, artist: editArtist, musicalKey: editKey, lyrics: editLyrics, updatedAt: Date.now() };
    try {
      if (activeSongId) await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'songs', activeSongId), data);
      else await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'songs'), data);
      setCurrentView('list'); showToast("¡Guardada!");
    } catch (e) { showToast("Error al guardar", "error"); }
    finally { setSaving(false); }
  };

  const formatInternetChords = () => {
    if (!editLyrics) return;
    const lines = editLyrics.split('\n');
    let newLyrics = [];
    const chordRegex = /^([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?(m|maj|min|dim|aug|sus|add|[0-9]|[-+()/#])*$/i;
    const isChordLine = (line) => {
      const words = line.trim().split(/\s+/);
      if (words.length === 0 || line.trim() === "") return false;
      let count = 0;
      words.forEach(w => { if (chordRegex.test(w)) count++; });
      return count > 0 && count >= words.length * 0.5;
    };
    for (let i = 0; i < lines.length; i++) {
      if (i < lines.length - 1 && isChordLine(lines[i]) && !isChordLine(lines[i+1])) {
        const chords = lines[i]; const text = lines[i+1];
        let merged = text.split('');
        const regexGlobal = /(([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?(?:m|maj|min|dim|aug|sus|add|[0-9]|[-+()/#])*)/gi;
        let match; let offset = 0;
        while ((match = regexGlobal.exec(chords)) !== null) {
          const pos = Math.min(match.index + offset, merged.length);
          const tag = `[${match[0]}]`;
          merged.splice(pos, 0, tag); offset += tag.length;
        }
        newLyrics.push(merged.join('')); i++;
      } else { newLyrics.push(lines[i]); }
    }
    setEditLyrics(newLyrics.join('\n')); showToast("Acordes vinculados");
  };

  const confirmDelete = async () => {
    try {
      if (itemToDelete.type === 'song') await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'songs', itemToDelete.id));
      else await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', itemToDelete.id));
      setItemToDelete(null); setCurrentView('list'); showToast("Eliminado");
    } catch (e) { showToast("Error al eliminar", "error"); }
  };

  // --- FILTROS DE BÚSQUEDA ---
  const filteredSongsList = songs.filter(s => 
    s.title.toLowerCase().includes(searchQuery.toLowerCase()) || 
    (s.artist && s.artist.toLowerCase().includes(searchQuery.toLowerCase())) ||
    (s.musicalKey && s.musicalKey.toLowerCase().includes(searchQuery.toLowerCase()))
  );

  return (
    <div className={`min-h-screen transition-all ${isDarkMode ? 'bg-slate-950 text-white' : 'bg-slate-50 text-slate-900'} pb-24 relative overflow-x-hidden`}>
      <style>{`
        @media print {
          .print\\:hidden { display: none !important; }
          main { max-width: 100% !important; padding: 0 !important; margin: 0 !important; }
          body { background: white !important; color: black !important; -webkit-print-color-adjust: exact; }
          .song-page { page-break-after: always; padding: 1.5cm 0; border-bottom: 1px solid #eee; }
          h2, h3 { color: black !important; }
          @page { margin: 1cm; size: auto; }
        }
      `}</style>

      {/* TOAST NOTIFICACIONES */}
      {toast && <div className="fixed top-6 left-1/2 -translate-x-1/2 px-6 py-3 rounded-full font-bold shadow-2xl z-[100] animate-in slide-in-from-top-4 bg-indigo-600 text-white print:hidden">{toast.message}</div>}

      {/* HEADER */}
      <header className={`sticky top-0 z-40 p-4 print:hidden ${isDarkMode ? 'bg-slate-900 border-b border-slate-800' : 'bg-indigo-700 text-white shadow-lg'}`}>
        <div className="max-w-4xl mx-auto flex justify-between items-center">
          <div className="flex items-center gap-2 cursor-pointer" onClick={() => {setCurrentView('list'); setIsScrolling(false)}}>
            <Music className="w-8 h-8 text-indigo-300" />
            <h1 className="text-2xl font-black italic tracking-tighter">CORARIO AI</h1>
          </div>
          <div className="flex items-center gap-3">
            <button onClick={() => setIsDarkMode(!isDarkMode)} className="p-2 hover:bg-white/20 rounded-full transition-colors">{isDarkMode ? <Sun /> : <Moon />}</button>
            {currentView === 'list' && (
              <button onClick={() => {setActiveSongId(null); setEditTitle(''); setEditLyrics(''); setEditArtist(''); setEditKey(''); setCurrentView('edit')}} className="bg-white text-indigo-700 px-4 py-2 rounded-full font-bold shadow-md hover:scale-105 active:scale-95 transition-all">
                <Plus className="w-4 h-4" /> Nueva
              </button>
            )}
          </div>
        </div>
      </header>

      <main className="max-w-4xl mx-auto p-4">
        
        {/* VISTA 1: LISTA PRINCIPAL */}
        {currentView === 'list' && (
          <div className="animate-in fade-in slide-in-from-bottom-4">
            <div className="flex flex-col sm:flex-row gap-4 mb-8">
              <div className="flex gap-2 p-1 bg-slate-200 dark:bg-slate-800 rounded-xl">
                <button onClick={() => setActiveTab('songs')} className={`flex-1 sm:flex-none px-6 py-2.5 rounded-lg font-bold transition ${activeTab === 'songs' ? 'bg-white dark:bg-slate-600 shadow-sm text-indigo-600 dark:text-white' : 'text-slate-500'}`}>Canciones</button>
                <button onClick={() => setActiveTab('playlists')} className={`flex-1 sm:flex-none px-6 py-2.5 rounded-lg font-bold transition ${activeTab === 'playlists' ? 'bg-white dark:bg-slate-600 shadow-sm text-indigo-600 dark:text-white' : 'text-slate-500'}`}>Cultos</button>
              </div>
              <div className="relative flex-1">
                <Search className="absolute left-3 top-3.5 w-5 h-5 opacity-40" />
                <input type="text" placeholder="Buscar título, tono o artista..." value={searchQuery} onChange={e => setSearchQuery(e.target.value)} className={`w-full pl-10 pr-4 py-3 rounded-xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 transition-colors shadow-sm ${isDarkMode ? 'border-slate-700' : 'border-slate-200'}`} />
              </div>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {activeTab === 'songs' ? (
                filteredSongsList.map(s => (
                  <div key={s.id} onClick={() => { setActiveSongId(s.id); setTransposeSteps(0); setAiAnalysis(null); setCurrentView('view'); }} className={`p-5 rounded-2xl border-2 transition-all cursor-pointer hover:border-indigo-500 flex justify-between items-center ${isDarkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-200 shadow-sm'}`}>
                    <div className="overflow-hidden pr-2">
                      <h3 className="text-lg font-bold truncate">{s.title}</h3>
                      <p className="text-xs opacity-50 truncate">{s.musicalKey && <span className="text-indigo-500 font-bold mr-2">[{s.musicalKey}]</span>}{s.artist || 'Artista desconocido'}</p>
                    </div>
                    <ChevronLeft className="rotate-180 opacity-20 shrink-0" />
                  </div>
                ))
              ) : (
                <>
                  <button onClick={() => setShowNewPlaylistModal(true)} className="p-8 rounded-2xl border-2 border-dashed border-slate-300 dark:border-slate-700 flex flex-col items-center justify-center gap-2 text-slate-500 hover:bg-indigo-50 dark:hover:bg-indigo-900/10 transition-all">
                    <FolderPlus /> <span className="font-bold">Crear Nuevo Culto</span>
                  </button>
                  {playlists.map(p => (
                    <div key={p.id} className={`rounded-2xl border-2 transition-all hover:border-indigo-500 overflow-hidden ${isDarkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-200 shadow-sm'}`}>
                      <div className="p-5 cursor-pointer" onClick={() => { setActivePlaylistId(p.id); setPlaylistMode('list'); setCurrentView('playlistView'); }}>
                        <h3 className="text-xl font-bold flex items-center gap-2"><ListMusic className="text-indigo-500"/> {p.name}</h3>
                        <p className="text-sm opacity-50">{p.songs.length} canciones preparadas</p>
                      </div>
                      <div className="px-5 py-3 border-t dark:border-slate-800 flex justify-end">
                        <button onClick={() => setItemToDelete({type:'playlist', id: p.id})} className="text-red-500 p-1 hover:bg-red-50 rounded transition-colors"><Trash2 className="w-4 h-4"/></button>
                      </div>
                    </div>
                  ))}
                </>
              )}
            </div>
          </div>
        )}

        {/* VISTA 2: CULTO / PLAYLIST */}
        {currentView === 'playlistView' && activePlaylist && (
          <div className="animate-in slide-in-from-right-4 duration-300">
            <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-8 gap-4 border-b dark:border-slate-800 pb-4 print:hidden">
              <div>
                <button onClick={() => {setCurrentView('list'); setIsScrolling(false)}} className="flex items-center text-indigo-500 font-bold mb-2 hover:underline"><ChevronLeft className="w-4 h-4"/> Volver</button>
                <h2 className="text-3xl font-black">{activePlaylist.name}</h2>
              </div>
              <div className="flex gap-2 w-full sm:w-auto">
                <button onClick={() => { setPlaylistMode(playlistMode === 'list' ? 'full' : 'list'); setIsScrolling(false); }} className={`flex-1 sm:flex-none px-6 py-3 rounded-xl font-bold flex items-center justify-center gap-2 transition-all shadow-lg ${playlistMode === 'full' ? 'bg-indigo-600 text-white' : 'bg-indigo-100 text-indigo-700'}`}>
                  {playlistMode === 'list' ? <Eye className="w-5 h-5"/> : <LayoutList className="w-5 h-5"/>}
                  {playlistMode === 'list' ? "Modo Presentación" : "Ver Lista"}
                </button>
                <button onClick={()=>setShowAddSongToCurrentPlaylistModal(true)} className="p-3 bg-indigo-600 text-white rounded-xl shadow-lg hover:bg-indigo-700 transition-colors"><Plus/></button>
                <button onClick={handlePrint} className="p-3 bg-slate-800 text-white rounded-xl shadow-lg hover:bg-slate-700 transition-colors"><Printer/></button>
              </div>
            </div>

            {playlistMode === 'list' ? (
              <div className="space-y-3 print:hidden">
                {activePlaylist.songs.map((sid, idx) => {
                  const s = songs.find(x => x.id === sid);
                  if (!s) return null;
                  return (
                    <div key={idx} className={`flex items-center p-4 rounded-xl border-2 ${isDarkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-200 shadow-sm'}`}>
                      <div className="w-8 h-8 rounded-full bg-indigo-100 dark:bg-indigo-900 text-indigo-500 flex items-center justify-center font-bold mr-4 shrink-0">{idx+1}</div>
                      <div className="flex-1 cursor-pointer truncate pr-4" onClick={() => { setActiveSongId(s.id); setCurrentView('view'); }}>
                        <h4 className="font-bold">{s.title}</h4>
                        <p className="text-xs opacity-50 truncate">{s.musicalKey && `[${s.musicalKey}] `}{s.artist}</p>
                      </div>
                      <button onClick={async () => {
                        const news = activePlaylist.songs.filter(id => id !== s.id);
                        await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', activePlaylist.id), { songs: news });
                      }} className="p-2 text-red-500 hover:bg-red-50 rounded-lg shrink-0"><X className="w-4 h-4"/></button>
                    </div>
                  );
                })}
              </div>
            ) : (
              <div className="space-y-24 pb-40">
                {activePlaylist.songs.map((sid, idx) => {
                  const s = songs.find(x => x.id === sid);
                  if (!s) return null;
                  return (
                    <div key={idx} className="song-page animate-in fade-in duration-500">
                      <div className="flex items-baseline gap-4 mb-6 border-b-2 border-indigo-500/20 pb-2">
                        <span className="text-6xl font-black opacity-10">{idx + 1}</span>
                        <div>
                          <h3 className={`text-4xl font-black ${fontFamily}`}>{s.title}</h3>
                          <p className="text-xl opacity-60 italic">{s.musicalKey ? `Tono: ${s.musicalKey}` : 'Sin escala'} • {s.artist}</p>
                        </div>
                      </div>
                      <div className="font-medium leading-relaxed">
                        {renderContent(s.lyrics, 0)} 
                      </div>
                    </div>
                  );
                })}
                <div className={`fixed bottom-6 right-6 flex items-center gap-2 p-2 rounded-full shadow-2xl z-40 print:hidden ${isScrolling ? 'bg-indigo-900/90 backdrop-blur' : 'bg-indigo-100/50 backdrop-blur border border-indigo-200'}`}>
                  {isScrolling && (
                    <div className="flex items-center pr-2">
                      <button onClick={() => setScrollSpeed(s => Math.max(1, s - 1))} className={`p-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'}`}><Rewind/></button>
                      <span className={`font-bold px-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'}`}>{scrollSpeed}x</span>
                      <button onClick={() => setScrollSpeed(s => Math.min(10, s + 1))} className={`p-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'}`}><FastForward/></button>
                    </div>
                  )}
                  <button onClick={() => setIsScrolling(!isScrolling)} className={`p-5 rounded-full shadow-xl transition-all ${isScrolling ? 'bg-red-500' : 'bg-indigo-600'} text-white hover:scale-105 active:scale-95`}>
                    {isScrolling ? <Pause className="w-8 h-8" /> : <Play className="w-8 h-8 ml-1" />}
                  </button>
                </div>
              </div>
            )}
          </div>
        )}

        {/* VISTA 3: LECTURA CANCIÓN INDIVIDUAL */}
        {currentView === 'view' && activeSong && (
          <div className="animate-in zoom-in-95 duration-300">
            <div className="flex justify-between items-center mb-8 print:hidden">
              <button onClick={() => { setIsMetronomePlaying(false); setIsScrolling(false); setShowStylePanel(false); setCurrentView('list'); }} className="flex items-center gap-1 text-indigo-500 font-bold hover:underline transition-all"><ChevronLeft /> Volver</button>
              <div className="flex gap-2">
                <button onClick={() => setShowAddToPlaylistModal(true)} className="p-3 bg-indigo-600 text-white rounded-xl shadow-lg hover:bg-indigo-700 transition-colors"><ListPlus className="w-5 h-5"/></button>
                <button onClick={generateDevotional} className="p-3 bg-amber-100 text-amber-700 rounded-xl shadow-lg hover:bg-amber-200 transition-colors">{isAiLoading ? <Loader2 className="animate-spin w-5 h-5"/> : <BookOpen className="w-5 h-5"/>}</button>
                <button onClick={() => { setEditTitle(activeSong.title); setEditLyrics(activeSong.lyrics); setEditKey(activeSong.musicalKey || ''); setEditArtist(activeSong.artist || ''); setCurrentView('edit'); }} className={`p-3 rounded-xl border-2 dark:bg-slate-900 dark:border-slate-800 hover:border-indigo-500 transition-colors`}><Edit2 className="w-5 h-5"/></button>
              </div>
            </div>

            {aiAnalysis && (
              <div className="bg-amber-50 dark:bg-amber-900/20 border-l-4 border-amber-400 p-6 rounded-r-2xl mb-8 animate-in slide-in-from-left-4 print:hidden shadow-sm">
                <div className="flex justify-between mb-2">
                  <h4 className="font-black text-amber-800 dark:text-amber-400 text-xs flex items-center gap-2 tracking-widest uppercase"><Sparkles className="w-4 h-4"/> Inspiración</h4>
                  <X className="w-4 h-4 cursor-pointer opacity-50" onClick={() => setAiAnalysis(null)} />
                </div>
                <div className="whitespace-pre-line italic text-amber-900 dark:text-amber-100 text-sm leading-relaxed font-medium">{aiAnalysis}</div>
              </div>
            )}

            <div className="mb-4">
              <h2 className={`text-5xl font-black ${fontFamily}`}>{activeSong.title}</h2>
              <p className="mt-2 text-xl opacity-60 italic">{activeSong.artist}</p>
              <div className="mt-4 flex gap-2 print:hidden">
                 <span className="px-3 py-1 bg-indigo-100 dark:bg-indigo-900/50 text-indigo-700 dark:text-indigo-300 rounded-lg font-bold">
                  Tono: {transposeSteps === 0 ? activeSong.musicalKey : transposeChord(activeSong.musicalKey, transposeSteps)}
                </span>
              </div>
            </div>

            <div className={`flex flex-wrap items-center justify-between gap-3 mb-8 p-3 rounded-2xl border-2 print:hidden ${isDarkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-200'} shadow-sm relative`}>
              <div className="flex items-center gap-1 bg-slate-100 dark:bg-slate-800 p-1.5 rounded-xl shadow-inner">
                <button onClick={() => setTransposeSteps(s => s - 1)} className="w-10 h-10 rounded-lg font-black bg-white dark:bg-slate-900 shadow-sm active:scale-95 transition-transform">-</button>
                <span className="font-black text-indigo-500 w-10 text-center text-lg">{transposeSteps > 0 ? `+${transposeSteps}` : transposeSteps}</span>
                <button onClick={() => setTransposeSteps(s => s + 1)} className="w-10 h-10 rounded-lg font-black bg-white dark:bg-slate-900 shadow-sm active:scale-95 transition-transform">+</button>
              </div>

              <div className="flex items-center gap-1 bg-slate-100 dark:bg-slate-800 p-1.5 rounded-xl shadow-inner">
                <button onClick={() => setBpm(b => Math.max(40, b - 5))} className="w-8 h-8 font-black hover:bg-slate-200 dark:hover:bg-slate-700 rounded transition-colors">-</button>
                <span className="font-bold text-teal-600 text-sm w-16 text-center">{bpm} BPM</span>
                <button onClick={() => setBpm(b => Math.min(240, b + 5))} className="w-8 h-8 font-black hover:bg-slate-200 dark:hover:bg-slate-700 rounded transition-colors">+</button>
                <button onClick={() => setIsMetronomePlaying(!isMetronomePlaying)} className={`ml-2 p-2 rounded-lg transition-all ${isMetronomePlaying ? 'bg-red-500 text-white shadow-lg' : 'bg-teal-500 text-white'}`}>
                  {isMetronomePlaying ? <Pause className="w-4 h-4"/> : <Play className="w-4 h-4"/>}
                </button>
              </div>

              <div className="flex gap-2">
                <button onClick={() => setShowStylePanel(!showStylePanel)} className={`p-3 rounded-xl border-2 transition-all ${showStylePanel ? 'bg-indigo-600 text-white border-indigo-600' : 'hover:border-indigo-500'}`} title="Panel de Colores"><Palette className="w-5 h-5"/></button>
                <button onClick={handlePrint} className="p-3 rounded-xl bg-slate-800 text-white shadow hover:bg-slate-700 transition-colors"><Printer className="w-5 h-5"/></button>
              </div>

              {/* PANEL ESTILOS (PERSONALIZACIÓN) */}
              {showStylePanel && (
                <div className={`absolute top-full right-0 mt-3 p-5 rounded-3xl border-2 shadow-2xl z-50 w-72 ${isDarkMode ? 'bg-slate-800 border-slate-700' : 'bg-white border-slate-200'} animate-in slide-in-from-top-4`}>
                  <label className="text-[10px] font-black uppercase text-slate-400 block mb-3 tracking-widest">Tipografía</label>
                  <div className="grid grid-cols-3 gap-2 mb-5">
                    <button onClick={()=>setFontFamily('font-sans')} className={`p-2 rounded-lg text-xs ${fontFamily === 'font-sans' ? 'bg-indigo-500 text-white font-bold' : 'bg-slate-100 dark:bg-slate-900 transition-colors'}`}>Moderna</button>
                    <button onClick={()=>setFontFamily('font-serif')} className={`p-2 rounded-lg text-xs font-serif ${fontFamily === 'font-serif' ? 'bg-indigo-500 text-white font-bold' : 'bg-slate-100 dark:bg-slate-900 transition-colors'}`}>Serif</button>
                    <button onClick={()=>setFontFamily('font-mono')} className={`p-2 rounded-lg text-xs font-mono ${fontFamily === 'font-mono' ? 'bg-indigo-500 text-white font-bold' : 'bg-slate-100 dark:bg-slate-900 transition-colors'}`}>Máquina</button>
                  </div>
                  
                  <label className="text-[10px] font-black uppercase text-slate-400 block mb-3 tracking-widest">Color Letra</label>
                  <div className="flex gap-2 mb-5">
                    {['default', 'blue', 'emerald', 'rose', 'amber'].map(c => (
                      <button key={c} onClick={()=>setLyricColor(c)} className={`w-8 h-8 rounded-full border-2 ${lyricColor===c?'border-indigo-500 scale-110 shadow-lg':'border-transparent'} transition-all`} style={{ backgroundColor: c==='default'?(isDarkMode?'#e2e8f0':'#1e293b'):c==='blue'?'#3b82f6':c==='emerald'?'#10b981':c==='rose'?'#f43f5e':c==='amber'?'#f59e0b':'#ccc' }}></button>
                    ))}
                  </div>

                  <label className="text-[10px] font-black uppercase text-slate-400 block mb-3 tracking-widest">Color Acordes</label>
                  <div className="flex gap-2">
                    {['default', 'red', 'teal', 'violet', 'orange'].map(c => (
                      <button key={c} onClick={()=>setChordColor(c)} className={`w-8 h-8 rounded-full border-2 ${chordColor===c?'border-white ring-2 ring-indigo-500 scale-110 shadow-lg':'border-transparent'} transition-all`} style={{ backgroundColor: c==='default'?'#6366f1':c==='red'?'#ef4444':c==='teal'?'#14b8a6':c==='violet'?'#8b5cf6':c==='orange'?'#f97316':'#ccc' }}></button>
                    ))}
                  </div>
                </div>
              )}
            </div>

            <div className="font-medium pb-24 leading-relaxed">{renderContent(activeSong.lyrics)}</div>

            <div className={`fixed bottom-6 right-6 flex items-center gap-2 p-2 rounded-full shadow-2xl z-40 print:hidden ${isScrolling ? 'bg-indigo-900/90 backdrop-blur' : 'bg-indigo-100/50 backdrop-blur border border-indigo-200'}`}>
              {isScrolling && (
                <div className="flex items-center pr-2 animate-in slide-in-from-right-2">
                  <button onClick={() => setScrollSpeed(s => Math.max(1, s - 1))} className={`p-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'} hover:scale-110 transition-transform`}><Rewind /></button>
                  <span className={`font-bold px-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'}`}>{scrollSpeed}x</span>
                  <button onClick={() => setScrollSpeed(s => Math.min(10, s + 1))} className={`p-2 ${isDarkMode ? 'text-white' : 'text-indigo-900'} hover:scale-110 transition-transform`}><FastForward /></button>
                </div>
              )}
              <button onClick={() => setIsScrolling(!isScrolling)} className={`p-5 rounded-full shadow-xl transition-all ${isScrolling ? 'bg-red-500 hover:bg-red-600' : 'bg-indigo-600 hover:bg-indigo-700'} text-white hover:scale-110 active:scale-95`}>
                {isScrolling ? <Pause className="w-8 h-8" /> : <Play className="w-8 h-8 ml-1" />}
              </button>
            </div>
          </div>
        )}

        {/* VISTA 4: EDITOR */}
        {currentView === 'edit' && (
          <div className="animate-in slide-in-from-right-4 duration-300">
            <div className="flex flex-col sm:flex-row justify-between sm:items-center mb-8 gap-4">
              <h2 className="text-3xl font-black">Editor de Alabanza</h2>
              <div className="flex gap-2">
                <button onClick={() => setCurrentView('list')} className="px-4 py-2 font-bold opacity-50 hover:opacity-100 transition-opacity">Cancelar</button>
                <button onClick={reharmonizeWithAI} disabled={isAiLoading} className="bg-indigo-100 text-indigo-700 px-6 py-2 rounded-xl font-bold flex items-center gap-2 hover:bg-indigo-200 transition-all shadow-sm">
                  {isAiLoading ? <Loader2 className="animate-spin w-5 h-5" /> : <Sparkles className="w-5 h-5" />} IA Rearmonizar
                </button>
              </div>
            </div>
            <div className="space-y-4">
              <input type="text" value={editTitle} onChange={e => setEditTitle(e.target.value)} placeholder="Título de la alabanza..." className={`w-full p-4 rounded-2xl border-2 text-xl font-bold focus:border-indigo-500 outline-none transition-all shadow-sm ${isDarkMode ? 'bg-slate-900 border-slate-700' : 'bg-white'}`} />
              <div className="grid grid-cols-2 gap-4">
                <input type="text" value={editKey} onChange={e => setEditKey(e.target.value)} placeholder="Escala/Tono (Ej: Do Mayor)..." className={`w-full p-4 rounded-2xl border-2 font-bold outline-none focus:border-indigo-500 transition-all ${isDarkMode ? 'bg-slate-900 border-slate-700' : 'bg-white'}`} />
                <input type="text" value={editArtist} onChange={e => setEditArtist(e.target.value)} placeholder="Autor / Artista..." className={`w-full p-4 rounded-2xl border-2 font-bold outline-none focus:border-indigo-500 transition-all ${isDarkMode ? 'bg-slate-900 border-slate-700' : 'bg-white'}`} />
              </div>
              <button onClick={formatInternetChords} className="w-full py-4 bg-indigo-50 text-indigo-700 rounded-2xl font-bold flex items-center justify-center gap-2 border-2 border-indigo-100 hover:bg-indigo-100 transition-all shadow-sm">
                <Wand2 className="w-5 h-5"/> Vincular Acordes con Letra (Obligatorio)
              </button>
              <textarea value={editLyrics} onChange={e => setEditLyrics(e.target.value)} placeholder="Pega la letra con acordes aquí..." className={`w-full h-96 p-6 rounded-3xl border-2 font-mono text-sm outline-none focus:border-indigo-500 transition-all resize-none ${isDarkMode ? 'bg-slate-900 border-slate-700' : 'bg-white'}`} />
              <button onClick={saveSong} disabled={saving} className="w-full bg-indigo-600 text-white py-5 rounded-3xl font-black text-xl shadow-xl hover:bg-indigo-700 active:scale-[0.98] transition-all">
                {saving ? <Loader2 className="animate-spin mx-auto" /> : <><Save className="inline-block mr-2" /> GUARDAR EN LA NUBE</>}
              </button>
            </div>
          </div>
        )}
      </main>

      {/* MODALES ADICIONALES */}
      {showNewPlaylistModal && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-50">
          <div className={`${isDarkMode ? 'bg-slate-800' : 'bg-white'} rounded-3xl p-6 max-w-sm w-full shadow-2xl animate-in zoom-in-95`}>
            <h3 className="text-xl font-bold mb-4">Nuevo Culto / Servicio</h3>
            <input type="text" value={newPlaylistName} onChange={e => setNewPlaylistName(e.target.value)} className={`w-full p-4 rounded-xl border-2 mb-6 outline-none focus:border-indigo-500 ${isDarkMode ? 'bg-slate-900 border-slate-700' : 'bg-white'}`} placeholder="Ej: Servicio Domingo..." autoFocus />
            <div className="flex gap-3">
              <button onClick={() => setShowNewPlaylistModal(false)} className="flex-1 py-3 bg-slate-100 dark:bg-slate-700 rounded-xl font-bold">Cancelar</button>
              <button onClick={async () => {
                await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'playlists'), { name: newPlaylistName.trim(), songs: [], updatedAt: Date.now() });
                setShowNewPlaylistModal(false); setNewPlaylistName(''); showToast("Culto Creado");
              }} className="flex-1 py-3 bg-indigo-600 text-white rounded-xl font-bold">Crear</button>
            </div>
          </div>
        </div>
      )}

      {showAddSongToCurrentPlaylistModal && activePlaylist && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-50">
          <div className={`${isDarkMode ? 'bg-slate-800 text-white' : 'bg-white text-slate-900'} rounded-3xl p-6 max-w-md w-full shadow-2xl max-h-[80vh] flex flex-col animate-in zoom-in-95`}>
            <div className="flex justify-between items-center mb-4">
              <h3 className="text-xl font-bold truncate pr-4">Añadir a {activePlaylist.name}</h3>
              <X className="cursor-pointer opacity-50 shrink-0" onClick={()=>setShowAddSongToCurrentPlaylistModal(false)}/>
            </div>
            <div className="overflow-y-auto space-y-2 flex-1 pr-2">
              {songs.map(s => (
                <button key={s.id} onClick={async () => {
                  if (!activePlaylist.songs.includes(s.id)) {
                    await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', activePlaylist.id), { songs: [...activePlaylist.songs, s.id] });
                    showToast("¡Agregada!");
                  }
                }} className={`w-full p-4 rounded-xl border-2 text-left flex justify-between items-center transition-all ${activePlaylist.songs.includes(s.id) ? 'bg-green-50 border-green-200 dark:bg-green-900/20' : 'hover:bg-indigo-50 border-transparent dark:border-slate-700 shadow-sm'}`}>
                   <div className="overflow-hidden"><p className="font-bold truncate">{s.title}</p><p className="text-xs opacity-50">{s.musicalKey || 'Sin escala'}</p></div>
                   {activePlaylist.songs.includes(s.id) ? <X className="text-green-600 w-5 h-5"/> : <Plus className="text-indigo-600 w-5 h-5"/>}
                </button>
              ))}
            </div>
            <button onClick={()=>setShowAddSongToCurrentPlaylistModal(false)} className="w-full mt-4 py-4 bg-indigo-600 text-white rounded-xl font-bold">Hecho</button>
          </div>
        </div>
      )}

      {showAddToPlaylistModal && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-50">
          <div className={`${isDarkMode ? 'bg-slate-800' : 'bg-white'} rounded-3xl p-6 max-w-sm w-full shadow-2xl animate-in zoom-in-95`}>
            <h3 className="text-xl font-bold mb-4">Seleccionar Culto</h3>
            <div className="space-y-2 max-h-64 overflow-y-auto pr-2">
              {playlists.map(p => (
                <button key={p.id} onClick={async () => {
                  if (!p.songs.includes(activeSongId)) await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', p.id), { songs: [...p.songs, activeSongId] });
                  setShowAddToPlaylistModal(false); showToast("Alabanza Añadida");
                }} className="w-full p-4 rounded-xl border-2 text-left hover:bg-indigo-50 dark:hover:bg-indigo-900/20 font-bold transition-all">{p.name}</button>
              ))}
              {playlists.length === 0 && <p className="text-center py-4 opacity-50">No tienes cultos creados.</p>}
            </div>
            <button onClick={() => { setShowAddToPlaylistModal(false); setShowNewPlaylistModal(true); }} className="w-full py-3 mt-4 bg-indigo-100 text-indigo-700 rounded-xl font-bold transition-all">Crear Nuevo Culto</button>
          </div>
        </div>
      )}

      {itemToDelete && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-[70]">
          <div className={`${isDarkMode ? 'bg-slate-800 text-white' : 'bg-white text-slate-900'} rounded-3xl p-8 max-w-sm w-full shadow-2xl text-center animate-in zoom-in-95`}>
            <div className="w-16 h-16 bg-red-100 text-red-600 rounded-full flex items-center justify-center mx-auto mb-4 animate-pulse"><Trash2 className="w-8 h-8" /></div>
            <h3 className="text-2xl font-black mb-2 text-slate-900">¿Borrar?</h3>
            <div className="flex gap-3">
              <button onClick={() => setItemToDelete(null)} className="flex-1 py-4 bg-slate-100 dark:bg-slate-700 rounded-xl font-bold">No</button>
              <button onClick={confirmDelete} className="flex-1 py-4 bg-red-600 text-white rounded-xl font-bold shadow-lg shadow-red-100">Borrar</button>
            </div>
          </div>
        </div>
      )}

    </div>
  );
}
