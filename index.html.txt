    <script type="text/babel">
        const { useState, useEffect, useRef, useMemo } = React;
        const { 
            Music, Plus, Search, Moon, Sun, ChevronLeft, Edit2, Trash2, 
            Save, Palette, Printer, Play, Pause, Rewind, FastForward,

        function App() {
            // Aquí pegaremos toda la lógica de canciones, scroll, metrónomo e IA 
            // que perfeccionamos en los mensajes anteriores.

            const [textSize, setTextSize] = useState(18);
            const [transposeSteps, setTransposeSteps] = useState(0);
            const [showStylePanel, setShowStylePanel] = useState(false);
            const [fontFamily, setFontFamily] = useState('font-sans');
            const [lyricColor, setLyricColor] = useState('default');
            const [chordColor, setChordColor] = useState('default');

            const [isScrolling, setIsScrolling] = useState(false);
            const [scrollSpeed, setScrollSpeed] = useState(2);
            const [bpm, setBpm] = useState(120);
            const [isMetronomePlaying, setIsMetronomePlaying] = useState(false);
            const [isAiLoading, setIsAiLoading] = useState(false);

            const [editTitle, setEditTitle] = useState('');
            const [editArtist, setEditArtist] = useState('');
            const [editKey, setEditKey] = useState('');
            const [editLyrics, setEditLyrics] = useState('');

                    { id: '1', title: 'Cuan Grande es Él', artist: 'Hymn', musicalKey: 'Sol', lyrics: '[Sol]Señor mi Dios, al con[Do]templar los cielos\nEl [Sol]firmamento y las [Re]estrellas mil' },

                ];
                setSongs(demoSongs);
                setPlaylists([{ id: 'p1', name: 'Culto de Domingo', songs: ['1', '2'] }]);
            }, []);

            const activeSong = songs.find(s => s.id === activeSongId);
            const activePlaylist = playlists.find(p => p.id === activePlaylistId);

            const showToast = (msg) => {
                setToast(msg);
                setTimeout(() => setToast(null), 3000);
            };

            const transposeChord = (chordStr, steps) => {
                if (!chordStr || steps === 0) return chordStr;
                const notes = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
                const map = { 'Db':'C#', 'Eb':'D#', 'Gb':'F#', 'Ab':'G#', 'Bb':'A#', 'Do':'C', 'Re':'D', 'Mi':'E', 'Fa':'F', 'Sol':'G', 'La':'A', 'Si':'B' };
                return chordStr.replace(/([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?/gi, (m) => {
                    let note = m.charAt(0).toUpperCase() + m.slice(1).toLowerCase();
                    if (map[note]) note = map[note];
                    let idx = notes.indexOf(note);
                    if (idx === -1) return m;
                    let newIdx = (idx + steps) % 12;
                    if (newIdx < 0) newIdx += 12;
                    return notes[newIdx];
                });
            };

                if (!text) return null;
                return text.split('\n').map((line, i) => (
                    <div key={i} className={`flex flex-wrap items-end mb-1 ${fontFamily}`}>
                        {line.split(/(\[[^\]]+\])/g).map((part, j) => {
                            if (part.startsWith('[') && part.endsWith(']')) {
                                return <span key={j} className="text-indigo-600 font-bold text-sm block -mb-1 mr-1">{transposeChord(part.slice(1, -1), steps)}</span>;
                            }
                            return <span key={j} className="text-slate-800 whitespace-pre">{part}</span>;
                        })}
                    </div>
                ));
            };

                                    <button onClick={() => setActiveTab('playlists')} className={`flex-1 py-2 rounded-xl font-bold ${activeTab === 'playlists' ? 'bg-indigo-600 text-white' : 'bg-white border'}`}>Cultos</button>
                                </div>
                                
                                <div className="relative mb-6">
                                    <Search className="absolute left-3 top-3 opacity-30" size={20} />
                                    <input 
                                        type="text" 
                                        placeholder="Buscar alabanza..." 
                                        className="w-full pl-10 pr-4 py-3 rounded-2xl border-2 outline-none focus:border-indigo-500"
                                        value={searchQuery}
                                        onChange={(e) => setSearchQuery(e.target.value)}
                                    />
                                </div>

                                <div className="grid grid-cols-1 gap-3">
                                    {activeTab === 'songs' ? (
                                        songs.filter(s => s.title.toLowerCase().includes(searchQuery.toLowerCase())).map(s => (
                                            <div key={s.id} onClick={() => {setActiveSongId(s.id); setCurrentView('view')}} className="p-4 bg-white rounded-2xl border-2 hover:border-indigo-500 cursor-pointer flex justify-between items-center shadow-sm">
                                                <div>
                                                    <h3 className="font-bold text-lg">{s.title}</h3>
                                                    <p className="text-sm opacity-50">{s.artist} • [{s.musicalKey}]</p>
                                                </div>
                                                <ChevronLeft className="rotate-180 opacity-20" />
                                            </div>
                                        ))
                                    ) : (
                                        playlists.map(p => (
                                            <div key={p.id} onClick={() => {setActivePlaylistId(p.id); setCurrentView('playlistView')}} className="p-4 bg-white rounded-2xl border-2 hover:border-indigo-500 cursor-pointer shadow-sm">
                                                <h3 className="font-bold text-xl flex items-center gap-2"><ListMusic className="text-indigo-500" /> {p.name}</h3>

                                            </div>
                                        ))
                                    )}
                                </div>

                        )}

                        {currentView === 'view' && activeSong && (
                            <div className="animate-in pb-20">
                                <button onClick={() => setCurrentView('list')} className="text-indigo-600 font-bold flex items-center gap-1 mb-6 print:hidden"><ChevronLeft size={16}/> Volver</button>

                                        <button className="px-3 font-black" onClick={() => setTransposeSteps(s => s-1)}>-</button>

                                    {renderLyrics(activeSong.lyrics, transposeSteps)}
                                </div>

                        )}

                            <div className="fixed bottom-6 right-6 flex items-center gap-2 p-2 bg-white/80 backdrop-blur rounded-full shadow-2xl border print:hidden">
                                {isScrolling && <span className="font-bold px-2 text-indigo-600">{scrollSpeed}x</span>}
                                <button 
                                    onClick={() => setIsScrolling(!isScrolling)}
                                    className={`p-4 rounded-full text-white shadow-lg ${isScrolling ? 'bg-red-500' : 'bg-indigo-600'}`}
                                >
                                    {isScrolling ? <Pause /> : <Play />}
                                </button>
                            </div>
                        )}
                    </main>

                    {toast && (
                        <div className="fixed bottom-10 left-1/2 -translate-x-1/2 bg-slate-800 text-white px-6 py-2 rounded-full font-bold animate-bounce shadow-xl">
                            {toast}
                        </div>
                    )}
                </div>
            );
        }
