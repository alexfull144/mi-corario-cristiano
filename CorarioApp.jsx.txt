import React, { useState, useEffect, useRef, useMemo } from 'react';
import { 
  Music, Plus, Search, Moon, Sun, ChevronLeft, Edit2, Trash2, 
  Save, Palette, Printer, Play, Pause, Rewind, FastForward, 
  BookOpen, ListPlus, X, Sparkles, FolderPlus, ListMusic, Eye, LayoutList, Loader2, Wand2, Check, ChevronRight
} from 'lucide-react';

// --- CONFIGURACIÓN DE FIREBASE ---
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';
import { getFirestore, collection, onSnapshot, addDoc, updateDoc, deleteDoc, doc, getDoc } from 'firebase/firestore';

// Inicialización de Firebase con variables de entorno del sistema
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

export default function App() {
  // --- ESTADOS ---
  const [user, setUser] = useState(null);
  const [songs, setSongs] = useState([]);
  const [playlists, setPlaylists] = useState([]);
  const [currentView, setCurrentView] = useState('list'); 
  const [activeTab, setActiveTab] = useState('songs');
  const [activeSongId, setActiveSongId] = useState(null);
  const [activePlaylistId, setActivePlaylistId] = useState(null);
  const [searchQuery, setSearchQuery] = useState('');
  const [playlistMode, setPlaylistMode] = useState('list'); 
  const [isDarkMode, setIsDarkMode] = useState(false);
  
  // Personalización Visual
  const [textSize, setTextSize] = useState(20);
  const [fontFamily, setFontFamily] = useState('font-sans');
  const [lyricColor, setLyricColor] = useState('default');
  const [chordColor, setChordColor] = useState('default');
  const [showStylePanel, setShowStylePanel] = useState(false);

  // Herramientas (Scroll, Metrónomo, IA)
  const [isScrolling, setIsScrolling] = useState(false);
  const [scrollSpeed, setScrollSpeed] = useState(2);
  const [bpm, setBpm] = useState(120);
  const [isMetronomePlaying, setIsMetronomePlaying] = useState(false);
  const [transposeSteps, setTransposeSteps] = useState(0);
  const [isAiLoading, setIsAiLoading] = useState(false);
  const [aiAnalysis, setAiAnalysis] = useState(null);
  const [toast, setToast] = useState(null);

  // Modales
  const [showAddSongModal, setShowAddSongModal] = useState(false);
  const [showNewPlaylistModal, setShowNewPlaylistModal] = useState(false);
  const [newPlaylistName, setNewPlaylistName] = useState('');

  // Editor
  const [editTitle, setEditTitle] = useState('');
  const [editArtist, setEditArtist] = useState('');
  const [editKey, setEditKey] = useState('');
  const [editLyrics, setEditLyrics] = useState('');

  const audioCtxRef = useRef(null);
  const metronomeIntervalRef = useRef(null);

  // --- AUTENTICACIÓN (Regla 3) ---
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // --- CARGA DE DATOS (Regla 1) ---
  useEffect(() => {
    if (!user) return;
    
    // Alabanzas del usuario
    const songsCol = collection(db, 'artifacts', appId, 'users', user.uid, 'songs');
    const unsubSongs = onSnapshot(songsCol, (snap) => {
      setSongs(snap.docs.map(d => ({ id: d.id, ...d.data() })));
    }, (err) => console.error("Error Firestore Songs:", err));

    // Cultos del usuario
    const playlistsCol = collection(db, 'artifacts', appId, 'users', user.uid, 'playlists');
    const unsubPlaylists = onSnapshot(playlistsCol, (snap) => {
      setPlaylists(snap.docs.map(d => ({ id: d.id, ...d.data() })));
    }, (err) => console.error("Error Firestore Playlists:", err));

    return () => { unsubSongs(); unsubPlaylists(); };
  }, [user]);

  // --- LÓGICA MUSICAL ---
  const transposeChord = (chordStr, steps) => {
    if (!chordStr || steps === 0) return chordStr;
    const notes = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
    const esNotes = ['Do', 'Do#', 'Re', 'Re#', 'Mi', 'Fa', 'Fa#', 'Sol', 'Sol#', 'La', 'La#', 'Si'];
    const map = { 'Db':'C#', 'Eb':'D#', 'Gb':'F#', 'Ab':'G#', 'Bb':'A#', 'Reb':'Do#', 'Mib':'Re#', 'Solb':'Fa#', 'Lab':'Sol#', 'Sib':'La#' };
    
    return chordStr.replace(/([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?/gi, (m) => {
      let note = m.charAt(0).toUpperCase() + m.slice(1).toLowerCase();
      if (map[note]) note = map[note];
      let idx = notes.indexOf(note);
      let isEs = false;
      if (idx === -1) { idx = esNotes.indexOf(note); isEs = true; }
      if (idx === -1) return m;
      let newIdx = (idx + steps) % 12;
      if (newIdx < 0) newIdx += 12;
      return isEs ? esNotes[newIdx] : notes[newIdx];
    });
  };

  const lyricStyleMap = { 
    'default': isDarkMode ? '#e2e8f0' : '#1e293b', 
    'blue': '#2563eb', 'emerald': '#059669', 'rose': '#e11d48', 'amber': '#d97706' 
  };
  
  const chordStyleMap = { 
    'default': '#4f46e5', 
    'red': '#dc2626', 'teal': '#0d9488', 'violet': '#7c3aed', 'orange': '#ea580c' 
  };

  const renderLyrics = (text, steps = 0) => {
    if (!text) return null;
    return text.split('\n').map((line, i) => (
      <div key={i} className={`flex flex-wrap items-end mb-1 ${fontFamily}`}>
        {line.split(/(\[[^\]]+\])/g).map((part, j) => {
          if (part.startsWith('[') && part.endsWith(']')) {
            return (
              <div key={j} className="flex flex-col mr-1">
                <span className="font-bold block -mb-1 print:text-blue-800" style={{ fontSize: `${textSize * 0.7}px`, color: chordStyleMap[chordColor] }}>
                  {transposeChord(part.slice(1, -1), steps)}
                </span>
              </div>
            );
          }
          return <span key={j} className="print:text-black whitespace-pre leading-relaxed" style={{ fontSize: `${textSize}px`, color: lyricStyleMap[lyricColor] }}>{part}</span>;
        })}
      </div>
    ));
  };

  // --- IA GEMINI (TU CLAVE) ---
  const askGemini = async (prompt) => {
    if (!user) return;
    setIsAiLoading(true);
    const apiKey = "gen-lang-client-0332007452";
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
      });
      const data = await response.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text || null;
    } catch (e) { 
      showToast("Error de IA", "error");
      return null; 
    } finally { 
      setIsAiLoading(false); 
    }
  };

  const handleRearmonizar = async () => {
    const r = await askGemini(`ACTÚA COMO UN PROCESADOR DE TEXTO PURO. Re-armoniza con acordes de piano avanzados entre corchetes [Acorde]. SIN COMENTARIOS. Letra: \n${editLyrics}`);
    if (r) { 
      setEditLyrics(r.replace(/```(text|markdown)?/g, '').replace(/```/g, '').trim()); 
      showToast("¡Rearmonización lista!"); 
    }
  };

  // --- FORMATO DE ACORDES INTERNET ---
  const formatInternetChords = () => {
    if (!editLyrics) return;
    const lines = editLyrics.split('\n');
    let newLyrics = [];
    const chordRegex = /^([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?(m|maj|min|dim|aug|sus|add|[0-9]|[-+()/#])*$/i;
    const isChordLine = (line) => {
      const words = line.trim().split(/\s+/);
      if (words.length === 0 || line.trim() === "") return false;
      let count = 0;
      words.forEach(w => { if (chordRegex.test(w)) count++; });
      return count > 0 && count >= words.length * 0.5;
    };
    for (let i = 0; i < lines.length; i++) {
      if (i < lines.length - 1 && isChordLine(lines[i]) && !isChordLine(lines[i+1])) {
        const chords = lines[i]; const text = lines[i+1];
        let merged = text.split('');
        const regexGlobal = /(([A-G]|Do|Re|Mi|Fa|Sol|La|Si)[#b]?(?:m|maj|min|dim|aug|sus|add|[0-9]|[-+()/#])*)/gi;
        let match; let offset = 0;
        while ((match = regexGlobal.exec(chords)) !== null) {
          const pos = Math.min(match.index + offset, merged.length);
          const tag = `[${match[0]}]`;
          merged.splice(pos, 0, tag); offset += tag.length;
        }
        newLyrics.push(merged.join('')); i++;
      } else { newLyrics.push(lines[i]); }
    }
    setEditLyrics(newLyrics.join('\n')); 
    showToast("Acordes vinculados");
  };

  // --- METRÓNOMO ---
  const playTick = () => {
    if (!audioCtxRef.current) audioCtxRef.current = new (window.AudioContext || window.webkitAudioContext)();
    const osc = audioCtxRef.current.createOscillator();
    const gain = audioCtxRef.current.createGain();
    osc.connect(gain); gain.connect(audioCtxRef.current.destination);
    osc.frequency.value = 1000;
    gain.gain.setValueAtTime(0.2, audioCtxRef.current.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtxRef.current.currentTime + 0.05);
    osc.start(); osc.stop(audioCtxRef.current.currentTime + 0.05);
  };

  useEffect(() => {
    if (isMetronomePlaying) {
      metronomeIntervalRef.current = setInterval(playTick, 60000 / bpm);
    } else { 
      clearInterval(metronomeIntervalRef.current); 
    }
    return () => clearInterval(metronomeIntervalRef.current);
  }, [isMetronomePlaying, bpm]);

  // --- AUTO SCROLL ---
  useEffect(() => {
    let interval;
    if (isScrolling) {
      interval = setInterval(() => window.scrollBy(0, scrollSpeed), 50);
    }
    return () => clearInterval(interval);
  }, [isScrolling, scrollSpeed]);

  // --- FILTROS ---
  const activeSong = useMemo(() => songs.find(s => s.id === activeSongId), [songs, activeSongId]);
  const activePlaylist = useMemo(() => playlists.find(p => p.id === activePlaylistId), [playlists, activePlaylistId]);

  const filteredSongs = songs.filter(s => 
    s.title.toLowerCase().includes(searchQuery.toLowerCase()) || 
    (s.artist && s.artist.toLowerCase().includes(searchQuery.toLowerCase())) ||
    (s.musicalKey && s.musicalKey.toLowerCase().includes(searchQuery.toLowerCase()))
  );

  const showToast = (message) => {
    setToast(message);
    setTimeout(() => setToast(null), 3000);
  };

  const saveSong = async () => {
    if (!user || !editTitle) return;
    const data = { title: editTitle, artist: editArtist, musicalKey: editKey, lyrics: editLyrics, updatedAt: Date.now() };
    try {
      if (activeSongId) await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'songs', activeSongId), data);
      else await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'songs'), data);
      setCurrentView('list');
      showToast("¡Alabanza guardada!");
    } catch (e) { console.error(e); }
  };

  return (
    <div className={`min-h-screen transition-colors duration-300 ${isDarkMode ? 'bg-slate-950 text-white' : 'bg-slate-50 text-slate-900'} font-sans pb-24 overflow-x-hidden`}>
      <style>{`
        @media print {
          .print\\:hidden { display: none !important; }
          main { max-width: 100% !important; padding: 0 !important; margin: 0 !important; }
          body { background: white !important; color: black !important; }
          .song-page { page-break-after: always; padding: 1.5cm 0; border-bottom: 2px solid #eee; }
        }
      `}</style>

      {toast && (
        <div className="fixed top-6 left-1/2 -translate-x-1/2 bg-indigo-600 text-white px-6 py-3 rounded-full font-bold shadow-2xl z-[100] animate-bounce">
          {toast}
        </div>
      )}

      {/* HEADER */}
      <header className="sticky top-0 z-50 bg-indigo-700 text-white p-4 shadow-lg flex justify-between items-center print:hidden">
        <div className="flex items-center gap-2 cursor-pointer" onClick={() => {setCurrentView('list'); setIsScrolling(false)}}>
          <Music size={24} />
          <h1 className="text-xl font-black italic tracking-tighter leading-none uppercase">Corario AI</h1>
        </div>
        <div className="flex items-center gap-4">
          <button onClick={() => setIsDarkMode(!isDarkMode)} className="p-2 hover:bg-white/10 rounded-full">
            {isDarkMode ? <Sun size={20} /> : <Moon size={20} />}
          </button>
          {currentView === 'list' && (
            <button 
              className="bg-white text-indigo-700 px-4 py-1.5 rounded-full font-bold text-sm shadow-md hover:scale-105 transition-transform flex items-center gap-1" 
              onClick={() => { setEditTitle(''); setEditArtist(''); setEditLyrics(''); setEditKey(''); setActiveSongId(null); setCurrentView('edit'); }}
            >
              <Plus size={16}/> Nueva
            </button>
          )}
        </div>
      </header>

      <main className="max-w-4xl mx-auto p-4 md:p-8">
        {/* LISTA PRINCIPAL */}
        {currentView === 'list' && (
          <div className="animate-in fade-in">
            <div className="flex gap-2 p-1 bg-slate-200 dark:bg-slate-800 rounded-2xl mb-8">
              <button onClick={() => setActiveTab('songs')} className={`flex-1 py-3 rounded-xl font-bold transition-all ${activeTab === 'songs' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500'}`}>Alabanzas</button>
              <button onClick={() => setActiveTab('playlists')} className={`flex-1 py-3 rounded-xl font-bold transition-all ${activeTab === 'playlists' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500'}`}>Cultos</button>
            </div>

            <div className="relative mb-8">
              <Search className="absolute left-4 top-3.5 opacity-30" size={20} />
              <input 
                type="text" 
                placeholder="Buscar por título, artista o tono..."
                className="w-full pl-12 pr-4 py-3.5 rounded-2xl border-2 dark:bg-slate-900 border-transparent focus:border-indigo-500 outline-none shadow-sm transition-all"
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
              />
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {activeTab === 'songs' ? (
                filteredSongs.map(s => (
                  <div key={s.id} onClick={() => {setActiveSongId(s.id); setTransposeSteps(0); setAiAnalysis(null); setCurrentView('view')}} className="p-5 bg-white dark:bg-slate-900 rounded-3xl border-2 hover:border-indigo-500 cursor-pointer flex justify-between items-center shadow-sm transition-all group">
                    <div className="overflow-hidden pr-2">
                      <h3 className="font-bold text-lg truncate group-hover:text-indigo-600">{s.title}</h3>
                      <p className="text-xs opacity-50 uppercase">{s.artist || 'Sin Autor'} • [{s.musicalKey || 'Tono'}]</p>
                    </div>
                    <ChevronRight className="opacity-20 group-hover:opacity-100 transition-opacity" />
                  </div>
                ))
              ) : (
                <>
                  <button onClick={() => setShowNewPlaylistModal(true)} className="p-8 rounded-3xl border-2 border-dashed border-slate-300 dark:border-slate-700 flex flex-col items-center justify-center gap-2 text-slate-400 hover:bg-indigo-50 dark:hover:bg-indigo-900/10 transition-all">
                    <FolderPlus size={32} />
                    <span className="font-bold">Crear Nuevo Culto</span>
                  </button>
                  {playlists.map(p => (
                    <div key={p.id} className="p-6 bg-white dark:bg-slate-900 rounded-3xl border-2 hover:border-indigo-500 cursor-pointer shadow-sm group">
                      <div className="flex items-center gap-3 mb-2" onClick={() => {setActivePlaylistId(p.id); setPlaylistMode('list'); setCurrentView('playlistView')}}>
                        <ListMusic size={24} className="text-indigo-500" />
                        <h3 className="font-black text-xl">{p.name}</h3>
                      </div>
                      <div className="flex justify-between items-center">
                         <p className="text-sm opacity-50">{p.songs?.length || 0} alabanzas</p>
                         <button 
                            onClick={() => deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', p.id))}
                            className="p-2 text-red-400 hover:text-red-600 hover:bg-red-50 rounded-lg transition-colors"
                         >
                            <Trash2 size={16}/>
                         </button>
                      </div>
                    </div>
                  ))}
                </>
              )}
            </div>
          </div>
        )}

        {/* LECTURA DE CANCIÓN */}
        {currentView === 'view' && activeSong && (
          <div className="animate-in zoom-in-95 duration-300">
            <div className="flex justify-between items-center mb-8 print:hidden">
              <button className="flex items-center gap-2 text-indigo-600 font-bold hover:underline" onClick={() => {setCurrentView('list'); setIsScrolling(false)}}>
                <ChevronLeft size={18} /> Volver
              </button>
              <div className="flex gap-2">
                <button className="p-3 bg-amber-100 text-amber-700 rounded-2xl hover:bg-amber-200 shadow-sm" onClick={async () => {
                  const res = await askGemini(`Genera un devocional cristiano breve para "${activeSong.title}". Incluye versículo, reflexión y oración corta.`);
                  if(res) setAiAnalysis(res);
                }}>
                   {isAiLoading ? <Loader2 className="animate-spin" size={20}/> : <BookOpen size={20}/>}
                </button>
                <button className="p-3 bg-indigo-100 text-indigo-700 rounded-2xl hover:bg-indigo-200" onClick={() => setShowStylePanel(!showStylePanel)}>
                  <Palette size={20}/>
                </button>
                <button className="p-3 bg-slate-800 text-white rounded-2xl" onClick={() => window.print()}>
                  <Printer size={20}/>
                </button>
                <button className="p-3 border-2 rounded-2xl" onClick={() => { setEditTitle(activeSong.title); setEditArtist(activeSong.artist); setEditLyrics(activeSong.lyrics); setEditKey(activeSong.musicalKey); setCurrentView('edit'); }}>
                  <Edit2 size={20}/>
                </button>
              </div>
            </div>

            {aiAnalysis && (
              <div className="bg-amber-50 dark:bg-amber-900/20 border-l-4 border-amber-400 p-6 rounded-r-2xl mb-8 animate-in slide-in-from-left-4 print:hidden shadow-sm relative">
                <button onClick={() => setAiAnalysis(null)} className="absolute top-4 right-4 text-amber-800"><X size={16}/></button>
                <h4 className="font-black text-amber-800 text-xs mb-2 flex items-center gap-1"><Sparkles size={14}/> INSPIRACIÓN PARA HOY</h4>
                <div className="italic text-amber-900 dark:text-amber-100 text-sm whitespace-pre-wrap">{aiAnalysis}</div>
              </div>
            )}

            {showStylePanel && (
              <div className="bg-white dark:bg-slate-900 p-6 rounded-3xl border-2 shadow-xl mb-8 animate-in print:hidden">
                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                  <div>
                    <label className="text-[10px] font-black uppercase text-slate-400 block mb-3 tracking-widest">Color Letra</label>
                    <div className="flex gap-2">
                      {['default', 'blue', 'emerald', 'rose', 'amber'].map(c => (
                        <button key={c} onClick={()=>setLyricColor(c)} className={`w-8 h-8 rounded-full border-2 ${lyricColor===c?'border-indigo-500 scale-110 shadow-md':'border-transparent'}`} style={{backgroundColor: c==='default' ? (isDarkMode ? '#e2e8f0' : '#1e293b') : c==='blue' ? '#3b82f6' : c==='emerald' ? '#10b981' : c==='rose' ? '#f43f5e' : '#f59e0b'}}></button>
                      ))}
                    </div>
                  </div>
                  <div>
                    <label className="text-[10px] font-black uppercase text-slate-400 block mb-3 tracking-widest">Color Acordes</label>
                    <div className="flex gap-2">
                      {['default', 'red', 'teal', 'violet', 'orange'].map(c => (
                        <button key={c} onClick={()=>setChordColor(c)} className={`w-8 h-8 rounded-full border-2 ${chordColor===c?'border-indigo-500 scale-110 shadow-md':'border-transparent'}`} style={{backgroundColor: c==='default' ? '#6366f1' : c==='red' ? '#ef4444' : c==='teal' ? '#14b8a6' : c==='violet' ? '#8b5cf6' : '#f97316'}}></button>
                      ))}
                    </div>
                  </div>
                </div>
              </div>
            )}

            <div className="mb-10">
              <h2 className={`text-5xl font-black mb-2 ${fontFamily}`}>{activeSong.title}</h2>
              <p className="text-xl opacity-60 italic mb-6">{activeSong.artist || 'Autor desconocido'}</p>
              
              <div className="flex flex-wrap gap-4 print:hidden">
                <div className="flex items-center bg-white dark:bg-slate-800 rounded-2xl border-2 p-1 shadow-sm">
                  <button className="w-10 h-10 font-black hover:bg-slate-100 dark:hover:bg-slate-700 rounded-xl" onClick={() => setTransposeSteps(s => s - 1)}>-</button>
                  <span className="px-4 text-indigo-600 font-black min-w-[100px] text-center">
                    {transposeSteps === 0 ? activeSong.musicalKey : transposeChord(activeSong.musicalKey, transposeSteps)}
                  </span>
                  <button className="w-10 h-10 font-black hover:bg-slate-100 dark:hover:bg-slate-700 rounded-xl" onClick={() => setTransposeSteps(s => s + 1)}>+</button>
                </div>
                <div className="flex items-center bg-white dark:bg-slate-800 rounded-2xl border-2 p-1 shadow-sm">
                  <button className="px-3 hover:bg-slate-100" onClick={() => setBpm(b => Math.max(40, b - 5))}>-</button>
                  <span className="font-bold text-xs w-16 text-center">{bpm} BPM</span>
                  <button className="px-3 hover:bg-slate-100" onClick={() => setBpm(b => Math.min(240, b + 5))}>+</button>
                  <button 
                    className={`ml-2 p-2 rounded-xl ${isMetronomePlaying?'bg-red-500':'bg-teal-500'} text-white shadow-md`} 
                    onClick={() => setIsMetronomePlaying(!isMetronomePlaying)}
                  >
                    {isMetronomePlaying ? <Pause size={16}/> : <Play size={16}/>}
                  </button>
                </div>
              </div>
            </div>
            <div className="space-y-1">{renderLyrics(activeSong.lyrics, transposeSteps)}</div>
          </div>
        )}

        {/* VISTA CULTO (MODO PRESENTACIÓN) */}
        {currentView === 'playlistView' && activePlaylist && (
          <div className="animate-in pb-40">
            <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-10 gap-4 border-b pb-6">
              <div>
                <button onClick={() => setCurrentView('list')} className="text-indigo-600 font-bold flex items-center gap-1 mb-2 hover:underline">
                  <ChevronLeft size={16}/> Volver
                </button>
                <h2 className="text-4xl font-black">{activePlaylist.name}</h2>
              </div>
              <div className="flex gap-2 w-full md:w-auto">
                <button 
                  onClick={() => { setPlaylistMode(playlistMode === 'list' ? 'full' : 'list'); setIsScrolling(false); }} 
                  className={`flex-1 md:flex-none px-6 py-3 rounded-2xl font-bold flex items-center justify-center gap-2 shadow-lg transition-all ${playlistMode === 'full' ? 'bg-indigo-600 text-white' : 'bg-indigo-100 text-indigo-700'}`}
                >
                  {playlistMode === 'list' ? <Eye size={20} /> : <LayoutList size={20} />} 
                  {playlistMode === 'list' ? "Modo Presentación" : "Ver Lista"}
                </button>
                <button onClick={()=>setShowAddSongModal(true)} className="p-3 bg-indigo-600 text-white rounded-2xl shadow-lg">
                  <Plus size={20}/>
                </button>
              </div>
            </div>
            
            {playlistMode === 'list' ? (
              <div className="space-y-4">
                {activePlaylist.songs?.map((sid, idx) => {
                  const s = songs.find(x => x.id === sid);
                  if (!s) return null;
                  return (
                    <div key={idx} className="flex items-center p-5 bg-white dark:bg-slate-900 rounded-3xl border-2 shadow-sm hover:border-indigo-200 transition-all group">
                      <div className="w-10 h-10 rounded-full bg-indigo-100 text-indigo-600 flex items-center justify-center font-black mr-4">{idx+1}</div>
                      <div className="flex-1 cursor-pointer" onClick={() => {setActiveSongId(s.id); setCurrentView('view')}}>
                        <h4 className="font-bold text-lg">{s.title}</h4>
                        <p className="text-sm opacity-50">{s.artist} • [{s.musicalKey}]</p>
                      </div>
                      <button 
                        onClick={()=>{
                          const news = activePlaylist.songs.filter(id => id !== sid);
                          updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', activePlaylist.id), { songs: news });
                        }} 
                        className="p-2 text-red-500 hover:bg-red-50 rounded-lg"
                      >
                        <Trash2 size={18}/>
                      </button>
                    </div>
                  );
                })}
              </div>
            ) : (
              <div className="space-y-40">
                {activePlaylist.songs?.map((sid, idx) => {
                  const s = songs.find(x => x.id === sid);
                  if (!s) return null;
                  return (
                    <div key={idx} className="song-page animate-in">
                      <div className="mb-10 border-b-4 border-indigo-600/20 pb-4">
                        <span className="text-7xl font-black text-indigo-600/10 block mb-2">{idx + 1}</span>
                        <h3 className={`text-4xl font-black ${fontFamily}`}>{s.title}</h3>
                        <p className="text-xl opacity-60 italic">{s.artist} • [{s.musicalKey}]</p>
                      </div>
                      <div className="text-2xl">{renderLyrics(s.lyrics, 0)}</div>
                    </div>
                  );
                })}
              </div>
            )}
          </div>
        )}

        {/* EDITOR */}
        {currentView === 'edit' && (
          <div className="animate-in pb-20">
            <div className="flex flex-col sm:flex-row justify-between sm:items-center mb-8 gap-4">
              <h2 className="text-3xl font-black uppercase tracking-tighter">Editor</h2>
              <div className="flex gap-2">
                <button onClick={() => setCurrentView('list')} className="px-4 py-2 font-bold opacity-50">Cerrar</button>
                <button onClick={handleRearmonizar} className="bg-indigo-100 text-indigo-700 px-6 py-2 rounded-xl font-bold flex items-center gap-2 hover:bg-indigo-200" disabled={isAiLoading}>
                  {isAiLoading ? <Loader2 size={18} className="animate-spin" /> : <Sparkles size={18} />} Rearmonizar IA
                </button>
              </div>
            </div>
            <div className="space-y-4">
              <input type="text" placeholder="Título de la Alabanza..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 text-xl font-bold shadow-sm" value={editTitle} onChange={e => setEditTitle(e.target.value)} />
              <div className="grid grid-cols-2 gap-4">
                <input type="text" placeholder="Compositor / Artista..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-bold shadow-sm" value={editArtist} onChange={e => setEditArtist(e.target.value)} />
                <input type="text" placeholder="Tono (Ej: Do Mayor)..." className="w-full p-4 rounded-2xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-bold shadow-sm" value={editKey} onChange={e => setEditKey(e.target.value)} />
              </div>
              <button onClick={formatInternetChords} className="w-full py-4 bg-indigo-50 text-indigo-700 rounded-2xl font-bold flex items-center justify-center gap-2 border-2 border-indigo-100 shadow-sm">
                <Wand2 size={20}/> Vincular Acordes (Paso Obligatorio)
              </button>
              <textarea placeholder="Pega letra con [Do] [Sol7] aquí..." className="w-full h-96 p-6 rounded-3xl border-2 dark:bg-slate-900 outline-none focus:border-indigo-500 font-mono text-sm leading-relaxed shadow-sm custom-scrollbar" value={editLyrics} onChange={e => setEditLyrics(e.target.value)} />
              <button className="w-full py-5 bg-indigo-600 text-white rounded-3xl font-black text-xl shadow-xl hover:bg-indigo-700 active:scale-95 transition-all" onClick={saveSong}>
                GUARDAR EN MI CORARIO
              </button>
            </div>
          </div>
        )}

        {/* CONTROLES FLOTANTES */}
        {(currentView === 'view' || (currentView === 'playlistView' && playlistMode === 'full')) && (
          <div className="fixed bottom-8 right-8 flex items-center gap-3 p-3 bg-white/90 dark:bg-slate-800/90 backdrop-blur rounded-full shadow-2xl border-2 print:hidden z-[100]">
            {isScrolling && (
              <div className="flex items-center gap-2 pr-2 animate-in slide-in-from-right-4">
                <button className="p-3 bg-slate-100 dark:bg-slate-700 rounded-full active:scale-90" onClick={() => setScrollSpeed(s => Math.max(1, s-1))}>
                  <Rewind size={18}/>
                </button>
                <span className="font-black text-indigo-600 px-1 text-sm">{scrollSpeed}x</span>
                <button className="p-3 bg-slate-100 dark:bg-slate-700 rounded-full active:scale-90" onClick={() => setScrollSpeed(s => Math.min(10, s+1))}>
                  <FastForward size={18}/>
                </button>
              </div>
            )}
            <button 
              onClick={() => setIsScrolling(!isScrolling)} 
              className={`p-6 rounded-full text-white shadow-xl transition-all active:scale-95 ${isScrolling ? 'bg-red-500' : 'bg-indigo-600'}`}
            >
              {isScrolling ? <Pause size={32} /> : <Play size={32} />}
            </button>
          </div>
        )}
      </main>
      
      {/* MODAL NUEVO CULTO */}
      {showNewPlaylistModal && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-[100] animate-in">
          <div className="bg-white dark:bg-slate-800 rounded-3xl p-6 max-w-sm w-full shadow-2xl">
            <h3 className="text-xl font-bold mb-4">Nombre del Culto / Servicio</h3>
            <input type="text" className="w-full p-4 rounded-2xl border-2 mb-6 dark:bg-slate-900 outline-none focus:border-indigo-500" placeholder="Ej: Domingo Mañana" value={newPlaylistName} onChange={e=>setNewPlaylistName(e.target.value)} autoFocus />
            <div className="flex gap-3">
              <button onClick={()=>setShowNewPlaylistModal(false)} className="flex-1 py-3 font-bold opacity-50">Cerrar</button>
              <button onClick={()=>{ if(newPlaylistName && user){ addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'playlists'), { name: newPlaylistName, songs: [], updatedAt: Date.now() }); setShowNewPlaylistModal(false); setNewPlaylistName(''); showToast("Culto Creado"); } }} className="bg-indigo-600 text-white px-6 rounded-xl font-bold">Crear</button>
            </div>
          </div>
        </div>
      )}

      {/* MODAL AÑADIR ALABANZA A CULTO */}
      {showAddSongModal && activePlaylist && (
        <div className="fixed inset-0 bg-black/80 flex items-center justify-center p-4 z-[100] animate-in">
          <div className="bg-white dark:bg-slate-800 rounded-3xl p-6 max-w-md w-full shadow-2xl max-h-[80vh] flex flex-col">
            <div className="flex justify-between items-center mb-4">
              <h3 className="text-xl font-bold truncate pr-4">Añadir a {activePlaylist.name}</h3>
              <button onClick={()=>setShowAddSongModal(false)} className="p-2 hover:bg-slate-100 rounded-full"><X size={20}/></button>
            </div>
            <div className="overflow-y-auto space-y-2 flex-1 pr-1 custom-scrollbar">
              {songs.map(s => (
                <button key={s.id} onClick={() => { if (!activePlaylist.songs?.includes(s.id)) { updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'playlists', activePlaylist.id), { songs: [...(activePlaylist.songs || []), s.id] }); showToast("¡Agregada!"); } }} className={`w-full p-4 rounded-xl border-2 text-left flex justify-between items-center transition-all ${activePlaylist.songs?.includes(s.id) ? 'bg-green-50 border-green-200 dark:bg-green-900/20' : 'hover:bg-indigo-50'}`}>
                  <div><p className="font-bold">{s.title}</p><p className="text-xs opacity-50">{s.artist}</p></div>
                  {activePlaylist.songs?.includes(s.id) ? <Check size={20} className="text-green-600" /> : <Plus size={20} className="text-indigo-600" />}
                </button>
              ))}
            </div>
            <button onClick={()=>setShowAddSongModal(false)} className="w-full mt-4 py-4 bg-indigo-600 text-white rounded-xl font-bold">Listo</button>
          </div>
        </div>
      )}
    </div>
  );
}
